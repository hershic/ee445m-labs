100 uint32_t
101 CPUUsageTick(void)
102 {
103  uint32_t ui32Value, ui32Usage;
104 
105  //
106  // Get the current value of the timer.
107  //
108  ui32Value =
109  MAP_TimerValueGet(g_pui32CPUUsageTimerBase[g_ui32CPUUsageTimer],
110  TIMER_A);
111 
112  //
113  // Based on the number of clock ticks accumulated by the timer during the
114  // previous timing period, compute the CPU usage as a 16.16 fixed-point
115  // value.
116  //
117  ui32Usage = ((((g_ui32CPUUsagePrevious - ui32Value) * 6400) /
118  g_ui32CPUUsageTicks) * 1024);
119 
120  //
121  // Save the previous value of the timer.
122  //
123  g_ui32CPUUsagePrevious = ui32Value;
124 
125  //
126  // Return the new CPU usage value.
127  //
128  return(ui32Usage);
129 }

146 void
147 CPUUsageInit(uint32_t ui32ClockRate, uint32_t ui32Rate, uint32_t ui32Timer)
148 {
149  //
150  // Check the arguments.
151  //
152  ASSERT(ui32ClockRate > ui32Rate);
153  ASSERT(ui32Timer < 6);
154 
155  //
156  // Save the timer index.
157  //
158  g_ui32CPUUsageTimer = ui32Timer;
159 
160  //
161  // Determine the number of system clocks per measurement period.
162  //
163  g_ui32CPUUsageTicks = ui32ClockRate / ui32Rate;
164 
165  //
166  // Set the previous value of the timer to the initial timer value.
167  //
168  g_ui32CPUUsagePrevious = 0xffffffff;
169 
170  //
171  // Enable peripheral clock gating.
172  //
173  MAP_SysCtlPeripheralClockGating(true);
174 
175  //
176  // Enable the third timer while the processor is in run mode, but disable
177  // it in sleep mode. It will therefore count system clocks when the
178  // processor is running but not when it is sleeping.
179  //
180  MAP_SysCtlPeripheralEnable(g_pui32CPUUsageTimerPeriph[ui32Timer]);
181  MAP_SysCtlPeripheralSleepDisable(g_pui32CPUUsageTimerPeriph[ui32Timer]);
182 
183  //
184  // Configure the third timer for 32-bit periodic operation.
185  //
186  MAP_TimerConfigure(g_pui32CPUUsageTimerBase[ui32Timer],
187  TIMER_CFG_PERIODIC);
188 
189  //
190  // Set the load value for the third timer to the maximum value.
191  //
192  MAP_TimerLoadSet(g_pui32CPUUsageTimerBase[ui32Timer], TIMER_A, 0xffffffff);
193 
194  //
195  // Enable the third timer. It will now count the number of system clocks
196  // during which the processor is executing code.
197  //
198  MAP_TimerEnable(g_pui32CPUUsageTimerBase[ui32Timer], TIMER_A);
199 }
