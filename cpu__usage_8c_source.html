00100 uint32_t
00101 CPUUsageTick(void)
00102 {
00103     uint32_t ui32Value, ui32Usage;
00104 
00105     //
00106     // Get the current value of the timer.
00107     //
00108     ui32Value =
00109         MAP_TimerValueGet(g_pui32CPUUsageTimerBase[g_ui32CPUUsageTimer],
00110                           TIMER_A);
00111 
00112     //
00113     // Based on the number of clock ticks accumulated by the timer during the
00114     // previous timing period, compute the CPU usage as a 16.16 fixed-point
00115     // value.
00116     //
00117     ui32Usage = ((((g_ui32CPUUsagePrevious - ui32Value) * 6400) /
00118                   g_ui32CPUUsageTicks) * 1024);
00119 
00120     //
00121     // Save the previous value of the timer.
00122     //
00123     g_ui32CPUUsagePrevious = ui32Value;
00124 
00125     //
00126     // Return the new CPU usage value.
00127     //
00128     return(ui32Usage);
00129 }

00146 void
00147 CPUUsageInit(uint32_t ui32ClockRate, uint32_t ui32Rate, uint32_t ui32Timer)
00148 {
00149     //
00150     // Check the arguments.
00151     //
00152     ASSERT(ui32ClockRate > ui32Rate);
00153     ASSERT(ui32Timer < 6);
00154 
00155     //
00156     // Save the timer index.
00157     //
00158     g_ui32CPUUsageTimer = ui32Timer;
00159 
00160     //
00161     // Determine the number of system clocks per measurement period.
00162     //
00163     g_ui32CPUUsageTicks = ui32ClockRate / ui32Rate;
00164 
00165     //
00166     // Set the previous value of the timer to the initial timer value.
00167     //
00168     g_ui32CPUUsagePrevious = 0xffffffff;
00169 
00170     //
00171     // Enable peripheral clock gating.
00172     //
00173     MAP_SysCtlPeripheralClockGating(true);
00174 
00175     //
00176     // Enable the third timer while the processor is in run mode, but disable
00177     // it in sleep mode.  It will therefore count system clocks when the
00178     // processor is running but not when it is sleeping.
00179     //
00180     MAP_SysCtlPeripheralEnable(g_pui32CPUUsageTimerPeriph[ui32Timer]);
00181     MAP_SysCtlPeripheralSleepDisable(g_pui32CPUUsageTimerPeriph[ui32Timer]);
00182 
00183     //
00184     // Configure the third timer for 32-bit periodic operation.
00185     //
00186     MAP_TimerConfigure(g_pui32CPUUsageTimerBase[ui32Timer],
00187                        TIMER_CFG_PERIODIC);
00188 
00189     //
00190     // Set the load value for the third timer to the maximum value.
00191     //
00192     MAP_TimerLoadSet(g_pui32CPUUsageTimerBase[ui32Timer], TIMER_A, 0xffffffff);
00193 
00194     //
00195     // Enable the third timer.  It will now count the number of system clocks
00196     // during which the processor is executing code.
00197     //
00198     MAP_TimerEnable(g_pui32CPUUsageTimerBase[ui32Timer], TIMER_A);
00199 }
