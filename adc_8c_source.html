00501 void
00502 ADCSequenceConfigure(uint32_t ui32Base, uint32_t ui32SequenceNum,
00503                      uint32_t ui32Trigger, uint32_t ui32Priority)
00504 {
00505     //
00506     // Check the arugments.
00507     //
00508     ASSERT((ui32Base == ADC0_BASE) || (ui32Base == ADC1_BASE));
00509     ASSERT(ui32SequenceNum < 4);
00510     ASSERT(((ui32Trigger & 0xF) == ADC_TRIGGER_PROCESSOR) ||
00511            ((ui32Trigger & 0xF) == ADC_TRIGGER_COMP0) ||
00512            ((ui32Trigger & 0xF) == ADC_TRIGGER_COMP1) ||
00513            ((ui32Trigger & 0xF) == ADC_TRIGGER_COMP2) ||
00514            ((ui32Trigger & 0xF) == ADC_TRIGGER_EXTERNAL) ||
00515            ((ui32Trigger & 0xF) == ADC_TRIGGER_TIMER) ||
00516            ((ui32Trigger & 0xF) == ADC_TRIGGER_PWM0) ||
00517            ((ui32Trigger & 0xF) == ADC_TRIGGER_PWM1) ||
00518            ((ui32Trigger & 0xF) == ADC_TRIGGER_PWM2) ||
00519            ((ui32Trigger & 0xF) == ADC_TRIGGER_PWM3) ||
00520            ((ui32Trigger & 0xF) == ADC_TRIGGER_ALWAYS) ||
00521            ((ui32Trigger & 0x30) == ADC_TRIGGER_PWM_MOD0) ||
00522            ((ui32Trigger & 0x30) == ADC_TRIGGER_PWM_MOD1));
00523     ASSERT(ui32Priority < 4);
00524 
00525     //
00526     // Compute the shift for the bits that control this sample sequence.
00527     //
00528     ui32SequenceNum *= 4;
00529 
00530     //
00531     // Set the trigger event for this sample sequence.
00532     //
00533     HWREG(ui32Base + ADC_O_EMUX) = ((HWREG(ui32Base + ADC_O_EMUX) &
00534                                      ~(0xf << ui32SequenceNum)) |
00535                                     ((ui32Trigger & 0xf) << ui32SequenceNum));
00536 
00537     //
00538     // Set the priority for this sample sequence.
00539     //
00540     HWREG(ui32Base + ADC_O_SSPRI) = ((HWREG(ui32Base + ADC_O_SSPRI) &
00541                                       ~(0xf << ui32SequenceNum)) |
00542                                      ((ui32Priority & 0x3) <<
00543                                       ui32SequenceNum));
00544 
00545     //
00546     // Set the source PWM module for this sequence's PWM triggers.
00547     //
00548     ui32SequenceNum *= 2;
00549     HWREG(ui32Base + ADC_O_TSSEL) = ((HWREG(ui32Base + ADC_O_TSSEL) &
00550                                       ~(0x30 << ui32SequenceNum)) |
00551                                      ((ui32Trigger & 0x30) <<
00552                                       ui32SequenceNum));
00553 }

00604 void
00605 ADCSequenceStepConfigure(uint32_t ui32Base, uint32_t ui32SequenceNum,
00606                          uint32_t ui32Step, uint32_t ui32Config)
00607 {
00608     uint32_t ui32Temp;
00609 
00610     //
00611     // Check the arguments.
00612     //
00613     ASSERT((ui32Base == ADC0_BASE) || (ui32Base == ADC1_BASE));
00614     ASSERT(ui32SequenceNum < 4);
00615     ASSERT(((ui32SequenceNum == 0) && (ui32Step < 8)) ||
00616            ((ui32SequenceNum == 1) && (ui32Step < 4)) ||
00617            ((ui32SequenceNum == 2) && (ui32Step < 4)) ||
00618            ((ui32SequenceNum == 3) && (ui32Step < 1)));
00619 
00620     //
00621     // Get the offset of the sequence to be configured.
00622     //
00623     ui32Base += ADC_SEQ + (ADC_SEQ_STEP * ui32SequenceNum);
00624 
00625     //
00626     // Compute the shift for the bits that control this step.
00627     //
00628     ui32Step *= 4;
00629 
00630     //
00631     // Set the analog mux value for this step.
00632     //
00633     HWREG(ui32Base + ADC_SSMUX) = ((HWREG(ui32Base + ADC_SSMUX) &
00634                                     ~(0x0000000f << ui32Step)) |
00635                                    ((ui32Config & 0x0f) << ui32Step));
00636 
00637     //
00638     // Set the upper bits of the analog mux value for this step.
00639     //
00640     HWREG(ui32Base + ADC_SSEMUX) = ((HWREG(ui32Base + ADC_SSEMUX) &
00641                                      ~(0x0000000f << ui32Step)) |
00642                                     (((ui32Config & 0xf00) >> 8) << ui32Step));
00643 
00644     //
00645     // Set the control value for this step.
00646     //
00647     HWREG(ui32Base + ADC_SSCTL) = ((HWREG(ui32Base + ADC_SSCTL) &
00648                                     ~(0x0000000f << ui32Step)) |
00649                                    (((ui32Config & 0xf0) >> 4) << ui32Step));
00650 
00651     //
00652     // Set the sample and hold time for this step.  This is not available on
00653     // all devices, however on devices that do not support this feature these
00654     // reserved bits are ignored on write access.
00655     //
00656     HWREG(ui32Base + ADC_SSTSH) = ((HWREG(ui32Base + ADC_SSTSH) &
00657                                     ~(0x0000000f << ui32Step)) |
00658                                 (((ui32Config & 0xf00000) >> 20) << ui32Step));
00659 
00660     //
00661     // Enable digital comparator if specified in the ui32Config bit-fields.
00662     //
00663     if(ui32Config & 0x000F0000)
00664     {
00665         //
00666         // Program the comparator for the specified step.
00667         //
00668         ui32Temp = HWREG(ui32Base + ADC_SSDC);
00669         ui32Temp &= ~(0xF << ui32Step);
00670         ui32Temp |= (((ui32Config & 0x00070000) >> 16) << ui32Step);
00671         HWREG(ui32Base + ADC_SSDC) = ui32Temp;
00672 
00673         //
00674         // Enable the comparator.
00675         //
00676         HWREG(ui32Base + ADC_SSOP) |= (1 << ui32Step);
00677     }
00678 
00679     //
00680     // Disable digital comparator if not specified.
00681     //
00682     else
00683     {
00684         HWREG(ui32Base + ADC_SSOP) &= ~(1 << ui32Step);
00685     }
00686 }

00824 ADCSequenceDataGet(uint32_t ui32Base, uint32_t ui32SequenceNum,
00825                    uint32_t *pui32Buffer)
00826 {
00827     uint32_t ui32Count;
00828 
00829     //
00830     // Check the arguments.
00831     //
00832     ASSERT((ui32Base == ADC0_BASE) || (ui32Base == ADC1_BASE));
00833     ASSERT(ui32SequenceNum < 4);
00834 
00835     //
00836     // Get the offset of the sequence to be read.
00837     //
00838     ui32Base += ADC_SEQ + (ADC_SEQ_STEP * ui32SequenceNum);
00839 
00840     //
00841     // Read samples from the FIFO until it is empty.
00842     //
00843     ui32Count = 0;
00844     while(!(HWREG(ui32Base + ADC_SSFSTAT) & ADC_SSFSTAT0_EMPTY) &&
00845           (ui32Count < 8))
00846     {
00847         //
00848         // Read the FIFO and copy it to the destination.
00849         //
00850         *pui32Buffer++ = HWREG(ui32Base + ADC_SSFIFO);
00851 
00852         //
00853         // Increment the count of samples read.
00854         //
00855         ui32Count++;
00856     }
00857 
00858     //
00859     // Return the number of samples read.
00860     //
00861     return(ui32Count);
00862 }

00967 void
00968 ADCSoftwareOversampleStepConfigure(uint32_t ui32Base, uint32_t ui32SequenceNum,
00969                                    uint32_t ui32Step, uint32_t ui32Config)
00970 {
00971     //
00972     // Check the arguments.
00973     //
00974     ASSERT((ui32Base == ADC0_BASE) || (ui32Base == ADC1_BASE));
00975     ASSERT(ui32SequenceNum < 3);
00976     ASSERT(((ui32SequenceNum == 0) &&
00977             (ui32Step < (8 >> g_pui8OversampleFactor[ui32SequenceNum]))) ||
00978            (ui32Step < (4 >> g_pui8OversampleFactor[ui32SequenceNum])));
00979 
00980     //
00981     // Get the offset of the sequence to be configured.
00982     //
00983     ui32Base += ADC_SEQ + (ADC_SEQ_STEP * ui32SequenceNum);
00984 
00985     //
00986     // Compute the shift for the bits that control this step.
00987     //
00988     ui32Step *= 4 << g_pui8OversampleFactor[ui32SequenceNum];
00989 
00990     //
00991     // Loop through the hardware steps that make up this step of the software
00992     // oversampled sequence.
00993     //
00994     for(ui32SequenceNum = 1 << g_pui8OversampleFactor[ui32SequenceNum];
00995         ui32SequenceNum; ui32SequenceNum--)
00996     {
00997         //
00998         // Set the analog mux value for this step.
00999         //
01000         HWREG(ui32Base + ADC_SSMUX) = ((HWREG(ui32Base + ADC_SSMUX) &
01001                                         ~(0x0000000f << ui32Step)) |
01002                                        ((ui32Config & 0x0f) << ui32Step));
01003 
01004         //
01005         // Set the upper bits of the analog mux value for this step.
01006         //
01007         HWREG(ui32Base + ADC_SSEMUX) = ((HWREG(ui32Base + ADC_SSEMUX) &
01008                                          ~(0x0000000f << ui32Step)) |
01009                                         (((ui32Config & 0xf00) >> 8) <<
01010                                          ui32Step));
01011 
01012         //
01013         // Set the control value for this step.
01014         //
01015         HWREG(ui32Base + ADC_SSCTL) = ((HWREG(ui32Base + ADC_SSCTL) &

01054                              uint32_t *pui32Buffer, uint32_t ui32Count)
01055 {
01056     uint32_t ui32Idx, ui32Accum;
01057 
01058     //
01059     // Check the arguments.
01060     //
01061     ASSERT((ui32Base == ADC0_BASE) || (ui32Base == ADC1_BASE));
01062     ASSERT(ui32SequenceNum < 3);
01063     ASSERT(((ui32SequenceNum == 0) &&
01064             (ui32Count < (8 >> g_pui8OversampleFactor[ui32SequenceNum]))) ||
01065            (ui32Count < (4 >> g_pui8OversampleFactor[ui32SequenceNum])));
01066 
01067     //
01068     // Get the offset of the sequence to be read.
01069     //
01070     ui32Base += ADC_SEQ + (ADC_SEQ_STEP * ui32SequenceNum);
01071 
01072     //
01073     // Read the samples from the FIFO until it is empty.
01074     //
01075     while(ui32Count--)

01077         //
01078         // Compute the sum of the samples.
01079         //

01086             //
01087             ui32Accum += HWREG(ui32Base + ADC_SSFIFO);
01088         }

01090         //
01091         // Write the averaged sample to the output buffer.
01092         //
01093         *pui32Buffer++ = ui32Accum >> g_pui8OversampleFactor[ui32SequenceNum];
