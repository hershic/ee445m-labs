00320 void
00321 MPURegionSet(uint32_t ui32Region, uint32_t ui32Addr, uint32_t ui32Flags)
00322 {
00323     //
00324     // Check the arguments.
00325     //
00326     ASSERT(ui32Region < 8);
00327     ASSERT(ui32Addr ==
00328            (ui32Addr & ~0 << (((ui32Flags & NVIC_MPU_ATTR_SIZE_M) >> 1) + 1)));
00329 
00330     //
00331     // Program the base address, use the region field to select the
00332     // region at the same time.
00333     //
00334     HWREG(NVIC_MPU_BASE) = ui32Addr | ui32Region | NVIC_MPU_BASE_VALID;
00335 
00336     //
00337     // Program the region attributes.  Set the TEX field and the S, C,
00338     // and B bits to fixed values that are suitable for all Tiva C and
00339     // E Series memory.
00340     //
00341     HWREG(NVIC_MPU_ATTR) = ((ui32Flags & ~(NVIC_MPU_ATTR_TEX_M |
00342                                            NVIC_MPU_ATTR_CACHEABLE)) |
00343                             NVIC_MPU_ATTR_SHAREABLE | NVIC_MPU_ATTR_BUFFRABLE);
00344 }
