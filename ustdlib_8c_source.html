00064 char *
00065 ustrncpy(char *  s1, const char *  s2, size_t n)
00066 {
00067     size_t count;
00068 
00069     //
00070     // Check the arguments.
00071     //
00072     ASSERT(s1);
00073     ASSERT(s2);
00074 
00075     //
00076     // Start at the beginning of the source string.
00077     //
00078     count = 0;
00079 
00080     //
00081     // Copy the source string until we run out of source characters or
00082     // destination space.
00083     //
00084     while(n && s2[count])
00085     {
00086         s1[count] = s2[count];
00087         count++;
00088         n--;
00089     }
00090 
00091     //
00092     // Pad the destination if we are not yet done.
00093     //
00094     while(n)
00095     {
00096         s1[count++] = (char)0;
00097         n--;
00098     }
00099 
00100     //
00101     // Pass the destination pointer back to the caller.
00102     //
00103     return(s1);
00104 }

00861 void
00862 ulocaltime(time_t timer, struct tm *tm)
00863 {
00864     time_t temp, months;
00865 
00866     //
00867     // Extract the number of seconds, converting time to the number of minutes.
00868     //
00869     temp = timer / 60;
00870     tm->tm_sec = timer - (temp * 60);
00871     timer = temp;
00872 
00873     //
00874     // Extract the number of minutes, converting time to the number of hours.
00875     //
00876     temp = timer / 60;
00877     tm->tm_min = timer - (temp * 60);
00878     timer = temp;
00879 
00880     //
00881     // Extract the number of hours, converting time to the number of days.
00882     //
00883     temp = timer / 24;
00884     tm->tm_hour = timer - (temp * 24);
00885     timer = temp;
00886 
00887     //
00888     // Compute the day of the week.
00889     //
00890     tm->tm_wday = (timer + 4) % 7;
00891 
00892     //
00893     // Compute the number of leap years that have occurred since 1968, the
00894     // first leap year before 1970.  For the beginning of a leap year, cut the
00895     // month loop below at March so that the leap day is classified as February
00896     // 29 followed by March 1, instead of March 1 followed by another March 1.
00897     //
00898     timer += 366 + 365;
00899     temp = timer / ((4 * 365) + 1);
00900     if((timer - (temp * ((4 * 365) + 1))) > (31 + 28))
00901     {
00902         temp++;
00903         months = 12;
00904     }
00905     else
00906     {
00907         months = 2;
00908     }
00909 
00910     //
00911     // Extract the year.
00912     //
00913     tm->tm_year = ((timer - temp) / 365) + 68;
00914     timer -= ((tm->tm_year - 68) * 365) + temp;
00915 
00916     //
00917     // Extract the month.
00918     //
00919     for(temp = 0; temp < months; temp++)
00920     {
00921         if(g_psDaysToMonth[temp] > timer)
00922         {
00923             break;
00924         }
00925     }
00926     tm->tm_mon = temp - 1;
00927 
00928     //
00929     // Extract the day of the month.
00930     //
00931     tm->tm_mday = timer - g_psDaysToMonth[temp - 1] + 1;
00932 }

00993     {
00994         return(1);
