02508 void
02509 GPIOPinConfigure(uint32_t ui32PinConfig)
02510 {
02511     uint32_t ui32Base, ui32Shift;
02512 
02513     //
02514     // Check the argument.
02515     //
02516     ASSERT(((ui32PinConfig >> 16) & 0xff) < 15);
02517     ASSERT(((ui32PinConfig >> 8) & 0xe3) == 0);
02518 
02519     //
02520     // Extract the base address index from the input value.
02521     //
02522     ui32Base = (ui32PinConfig >> 16) & 0xff;
02523 
02524     //
02525     // Get the base address of the GPIO module, selecting either the APB or the
02526     // AHB aperture as appropriate.
02527     //
02528     if(HWREG(SYSCTL_GPIOHBCTL) & (1 << ui32Base))
02529     {
02530         ui32Base = g_pui32GPIOBaseAddrs[(ui32Base << 1) + 1];
02531     }
02532     else
02533     {
02534         ui32Base = g_pui32GPIOBaseAddrs[ui32Base << 1];
02535     }
02536 
02537     //
02538     // Extract the shift from the input value.
02539     //
02540     ui32Shift = (ui32PinConfig >> 8) & 0xff;
02541 
02542     //
02543     // Write the requested pin muxing value for this GPIO pin.
02544     //
02545     HWREG(ui32Base + GPIO_O_PCTL) = ((HWREG(ui32Base + GPIO_O_PCTL) &
02546                                       ~(0xf << ui32Shift)) |
02547                                      ((ui32PinConfig & 0xf) << ui32Shift));
02548 }
