00339 EPIDividerCSSet(uint32_t ui32Base, uint32_t ui32CS,
00340                 uint32_t ui32Divider)
00341 {
00342     uint32_t ui32Reg;
00343 
00344     //
00345     // Check the arguments.
00346     //
00347     ASSERT(ui32Base == EPI0_BASE);
00348     ASSERT(ui32CS < 4);
00349 
00350     //
00351     // Write the divider value to the register bitfield.
00352     //
00353     if(ui32CS < 2)
00354     {
00355         ui32Reg = HWREG(ui32Base + EPI_O_BAUD) & ~(0xffff << (16 * ui32CS));
00356         ui32Reg |= ((ui32Divider & 0xffff) << (16 * ui32CS));
00357         HWREG(ui32Base + EPI_O_BAUD) = ui32Reg;
00358     }
00359     else
00360     {
00361         ui32Reg = (HWREG(ui32Base + EPI_O_BAUD2) &
00362                    ~(0xffff << (16 * (ui32CS - 2))));
00363         ui32Reg |= ((ui32Divider & 0xffff) << (16 * (ui32CS - 2)));
00364         HWREG(ui32Base + EPI_O_BAUD2) = ui32Reg;
00365     }
