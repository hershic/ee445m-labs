00133     // Check the arguments.
00134     //
00135     ASSERT(!(ui32Address & (FLASH_ERASE_SIZE - 1)));
00136 
00137     //
00138     // Clear the flash access and error interrupts.
00139     //
00140     HWREG(FLASH_FCMISC) = (FLASH_FCMISC_AMISC | FLASH_FCMISC_VOLTMISC |
00141                            FLASH_FCMISC_ERMISC);
00142 
00143     //
00144     // Erase the block.
00145     //
00146     HWREG(FLASH_FMA) = ui32Address;
00147     HWREG(FLASH_FMC) = FLASH_FMC_WRKEY | FLASH_FMC_ERASE;
00148 
00149     //
00150     // Wait until the block has been erased.
00151     //
00152     while(HWREG(FLASH_FMC) & FLASH_FMC_ERASE)

00192 FlashProgram(uint32_t *pui32Data, uint32_t ui32Address, uint32_t ui32Count)
00193 {
00194     //
00195     // Check the arguments.
00196     //
00197     ASSERT(!(ui32Address & 3));
00198     ASSERT(!(ui32Count & 3));
00199 
00200     //
00201     // Clear the flash access and error interrupts.
00202     //
00203     HWREG(FLASH_FCMISC) = (FLASH_FCMISC_AMISC | FLASH_FCMISC_VOLTMISC |
00204                            FLASH_FCMISC_INVDMISC | FLASH_FCMISC_PROGMISC);
00205 
00206     //
00207     // Loop over the words to be programmed.
00208     //
00209     while(ui32Count)
00210     {
00211         //
00212         // Set the address of this block of words.
00213         //
00214         HWREG(FLASH_FMA) = ui32Address & ~(0x7f);
00215 
00216         //
00217         // Loop over the words in this 32-word block.
00218         //
00219         while(((ui32Address & 0x7c) || (HWREG(FLASH_FWBVAL) == 0)) &&
00220               (ui32Count != 0))
00221         {
00222             //
00223             // Write this word into the write buffer.
00224             //
00225             HWREG(FLASH_FWBN + (ui32Address & 0x7c)) = *pui32Data++;
00226             ui32Address += 4;
00227             ui32Count -= 4;
00228         }
00229 
00230         //
00231         // Program the contents of the write buffer into flash.
00232         //
00233         HWREG(FLASH_FMC2) = FLASH_FMC2_WRKEY | FLASH_FMC2_WRBUF;
00234 
00235         //
00236         // Wait until the write buffer has been programmed.
00237         //
00238         while(HWREG(FLASH_FMC2) & FLASH_FMC2_WRBUF)
00239         {
00240         }
00241     }
00242 
00243     //
00244     // Return an error if an access violation occurred.
00245     //
00246     if(HWREG(FLASH_FCRIS) & (FLASH_FCRIS_ARIS | FLASH_FCRIS_VOLTRIS |
00247                              FLASH_FCRIS_INVDRIS | FLASH_FCRIS_PROGRIS))
00248     {
00249         return(-1);
00250     }
00251 
00252     //
00253     // Success.
00254     //
00255     return(0);
00256 }

00280 
00281     //
00282     // Check the argument.
00283     //
00284     ASSERT(!(ui32Address & (FLASH_PROTECT_SIZE - 1)));
00285

00287     // Calculate the Flash Bank from Base Address, and mask off the Bank
00288     // from ui32Address for subsequent reference.
00289     //
00290     ui32Bank = (((ui32Address / FLASH_PROTECT_SIZE) / 32) % 4);
00291     ui32Address &= ((FLASH_PROTECT_SIZE * 32) - 1);
00292 
00293     //
00294     // Read the appropriate flash protection registers for the specified
00295     // flash bank.
00296     //
00297     ui32FMPRE = HWREG(g_pui32FMPRERegs[ui32Bank]);
00298     ui32FMPPE = HWREG(g_pui32FMPPERegs[ui32Bank]);
00299 
00300     //
00301     // Check the appropriate protection bits for the block of memory that
00302     // is specified by the address.
00303     //
00304     switch((((ui32FMPRE >> (ui32Address / FLASH_PROTECT_SIZE)) & 0x1) << 1) |
00305            ((ui32FMPPE >> (ui32Address / FLASH_PROTECT_SIZE)) & 0x1))
00306     {
00307         //
00308         // This block is marked as execute only (that is, it can not be erased
00309         // or programmed, and the only reads allowed are via the instruction
00310         // fetch interface).
00311         //
00312         case 0:
00313         case 1:
00314         {
00315             return(FlashExecuteOnly);
00316         }
00317 
00318         //
00319         // This block is marked as read only (that is, it can not be erased or
00320         // programmed).
00321         //
00322         case 2:
00323         {
00324             return(FlashReadOnly);
00325         }
00326 
00327         //
00328         // This block is read/write; it can be read, erased, and programmed.
00329         //
00330         case 3:
00331         default:
00332         {
00333             return(FlashReadWrite);
00334         }
00335     }

00365 FlashProtectSet(uint32_t ui32Address, tFlashProtection eProtect)
00366 {
00367     uint32_t ui32ProtectRE, ui32ProtectPE;
00368     uint32_t ui32Bank;
00369 
00370     //
00371     // Check the argument.
00372     //
00373     ASSERT(!(ui32Address & (FLASH_PROTECT_SIZE - 1)));
00374     ASSERT((eProtect == FlashReadWrite) || (eProtect == FlashReadOnly) ||
00375            (eProtect == FlashExecuteOnly));
00376 
00377     //
00378     // Convert the address into a block number.
00379     //
00380     ui32Address /= FLASH_PROTECT_SIZE;
00381 
00382     //
00383     // ui32Address contains a "raw" block number.  Derive the Flash Bank from
00384     // the "raw" block number, and convert ui32Address to a "relative"
00385     // block number.
00386     //
00387     ui32Bank = ((ui32Address / 32) % 4);
00388     ui32Address %= 32;
00389 
00390     //
00391     // Get the current protection for the specified flash bank.
00392     //
00393     ui32ProtectRE = HWREG(g_pui32FMPRERegs[ui32Bank]);
00394     ui32ProtectPE = HWREG(g_pui32FMPPERegs[ui32Bank]);
00395 
00396     //
00397     // Set the protection based on the requested protection.
00398     //
00399     switch(eProtect)
00400     {
00401         //
00402         // Make this block execute only.
00403         //
00404         case FlashExecuteOnly:
00405         {
00406             //
00407             // Turn off the read and program bits for this block.
00408             //
00409             ui32ProtectRE &= ~(0x1 << ui32Address);
00410             ui32ProtectPE &= ~(0x1 << ui32Address);
00411 
00412             //
00413             // We're done handling this protection.
00414             //
00415             break;
00416         }
00417 
00418         //
00419         // Make this block read only.
00420         //
00421         case FlashReadOnly:
00422         {
00423             //
00424             // The block can not be made read only if it is execute only.
00425             //
00426             if(((ui32ProtectRE >> ui32Address) & 0x1) != 0x1)
00427             {
00428                 return(-1);
00429             }
00430 
00431             //
00432             // Make this block read only.
00433             //
00434             ui32ProtectPE &= ~(0x1 << ui32Address);
00435 
00436             //
00437             // We're done handling this protection.
00438             //
00439             break;
00440         }
00441 
00442         //
00443         // Make this block read/write.
00444         //
00445         case FlashReadWrite:
00446         default:
00447         {
00448             //
00449             // The block can not be made read/write if it is not already
00450             // read/write.
00451             //
00452             if((((ui32ProtectRE >> ui32Address) & 0x1) != 0x1) ||
00453                (((ui32ProtectPE >> ui32Address) & 0x1) != 0x1))
00454             {
00455                 return(-1);
00456             }
00457 
00458             //
00459             // The block is already read/write, so there is nothing to do.
00460             //
00461             return(0);
00462         }
00463     }
00464 
00465     //
00466     // Set the new protection for the specified flash bank.
00467     //
00468     HWREG(g_pui32FMPRERegs[ui32Bank]) = ui32ProtectRE;
00469     HWREG(g_pui32FMPPERegs[ui32Bank]) = ui32ProtectPE;
00470 
00471     //
00472     // Success.
00473     //
00474     return(0);
00475 }
