00225 HibernateClockConfig(uint32_t ui32Config)
00226 {
00227     uint32_t ui32HIBCtl;
00228 
00229     ASSERT((ui32Config & ~(HIBERNATE_OSC_HIGHDRIVE | HIBERNATE_OSC_LOWDRIVE |
00230                            HIBERNATE_OSC_DISABLE)) == 0);
00231 
00232     ui32HIBCtl = HWREG(HIB_CTL);
00233 
00234     //
00235     // Clear the current configuration bits.
00236     //
00237     ui32HIBCtl &= ~(HIBERNATE_OSC_HIGHDRIVE | HIBERNATE_OSC_LOWDRIVE |
00238                     HIBERNATE_OSC_LFIOSC | HIBERNATE_OSC_DISABLE);
00239 
00240     //
00241     // Set the new configuration bits.
00242     //
00243     ui32HIBCtl |= ui32Config & (HIBERNATE_OSC_HIGHDRIVE |
00244                                 HIBERNATE_OSC_LOWDRIVE |
00245                                 HIBERNATE_OSC_LFIOSC |
00246                                 HIBERNATE_OSC_DISABLE);
00247 
00248     //
00249     // Must be sure that the 32KHz clock is enabled if the hibernate is about
00250     // to switch to it.
00251     //
00252     if(ui32Config & HIBERNATE_OSC_LFIOSC)
00253     {
00254         ui32HIBCtl |= HIB_CTL_CLK32EN;
00255     }
00256 
00257     //
00258     // Set the hibernation clocking configuration.
00259     //
00260     HWREG(HIB_CTL) = ui32HIBCtl;
00261 
00262     //
00263     // Wait for write completion
00264     //
00265     _HibernateWriteComplete();
00266 
00267     //
00268     // Write the output clock configuration for devices that support
00269     // controlling the output clocks from the hibernate module.
00270     //
00271     if(HIBERNATE_CLOCK_OUTPUT)
00272     {
00273         HWREG(HIB_CC) = ui32Config & (HIBERNATE_OUT_SYSCLK |
00274                                       HIBERNATE_OUT_ALT1CLK);
00275     }
00276 }

00419 void
00420 HibernateWakeSet(uint32_t ui32WakeFlags)
00421 {
00422     //
00423     // Check the arguments.
00424     //
00425     ASSERT(!(ui32WakeFlags & ~(HIBERNATE_WAKE_PIN | HIBERNATE_WAKE_RTC |
00426                                HIBERNATE_WAKE_GPIO | HIBERNATE_WAKE_RESET |
00427                                HIBERNATE_WAKE_LOW_BAT)));
00428 
00429     //
00430     // Set the specified wake flags in the control register.
00431     //
00432     HWREG(HIB_CTL) = (ui32WakeFlags | (HWREG(HIB_CTL) &
00433                                        ~(HIBERNATE_WAKE_PIN |
00434                                          HIBERNATE_WAKE_RTC |
00435                                          HIBERNATE_WAKE_LOW_BAT)));
00436 
00437     //
00438     // Wait for write completion
00439     //
00440     _HibernateWriteComplete();
00441 
00442     //
00443     // Write the hibernate IO register if requested.
00444     //
00445     if(HIBERNATE_WAKE_IO)
00446     {
00447         //
00448         // If the reset or GPIOs are begin used as a wake source then the
00449         // the VDD3ON needs to be set to allow the pads to remained
00450         // powered.
00451         //
00452         if((ui32WakeFlags & (HIBERNATE_WAKE_RESET | HIBERNATE_WAKE_GPIO)) &&
00453            ((HWREG(HIB_CTL) & HIB_CTL_VDD3ON) == 0))
00454         {
00455             //
00456             // Make sure that VDD3ON mode is enabled so that the pads can
00457             // retain their state.
00458             //
00459             HWREG(HIB_CTL) |= HIB_CTL_VDD3ON;
00460 
00461             //
00462             // Wait for write completion
00463             //
00464             _HibernateWriteComplete();
00465         }
00466 
00467         //
00468         // Set the requested flags.
00469         //
00470         HWREG(HIB_IO) = (ui32WakeFlags >> 16) | HIB_IO_WUUNLK;
00471 
00472         //
00473         // Spin until the write complete bit is set.
00474         //
00475         while((HWREG(HIB_IO) & HIB_IO_IOWRC) == 0)
00476         {
00477         }
00478 
00479         //
00480         // Clear the write unlock bit.
00481         //
00482         HWREG(HIB_IO) &= ~HIB_IO_WUUNLK;
00483     }
00484 }

01459 //*****************************************************************************
01460 //
01461 // Internal function to parse the time structure to set the calendar fields.
01462 //
01463 //*****************************************************************************
01464 static void
01465 _HibernateCalendarSet(uint32_t ui32Reg, struct tm *psTime)
01466 {
01467     uint32_t ui32Time, ui32Date;
01468 
01469     ASSERT(HWREG(HIB_CALCTL) & HIB_CALCTL_CALEN);
01470 
01471     //
01472     // Minutes and seconds are consistent in all modes.
01473     //
01474     ui32Time = (((psTime->tm_min << HIB_CALLD0_MIN_S) & HIB_CALLD0_MIN_M) |
01475                 ((psTime->tm_sec << HIB_CALLD0_SEC_S) & HIB_CALLD0_SEC_M));
01476 
01477     //
01478     // 24 Hour time is used directly for Calendar set.
01479     //
01480     if(HWREG(HIB_CALCTL) & HIB_CALCTL_CAL24)
01481     {
01482         ui32Time |= (psTime->tm_hour << HIB_CALLD0_HR_S);
01483 
01484         //
01485         // for Calendar match, if it is every hour, AMPM bit should be clear
01486         //
01487         if((ui32Reg == HIB_CALM0) && (psTime->tm_hour == 0xFF) )
01488         {
01489             //
01490             // clear AMPM bit
01491             //
01492             ui32Time &= ~HIB_CAL0_AMPM;
01493         }
01494     }
01495     else
01496     {
01497         //
01498         // In AM/PM time hours have to be capped at 12.
01499         // If the hours are all 1s, it means the match for the hour is
01500         // always true.  We need to set 1F in the hw field.
01501         //
01502         if(psTime->tm_hour == 0xFF)
01503         {
01504             //
01505             // Match every hour.
01506             //
01507             ui32Time |= HIB_CALLD0_HR_M;
01508         }
01509         else if(psTime->tm_hour >= 12)
01510         {
01511             //
01512             // Need to set the PM bit if it is noon or later.
01513             //
01514             ui32Time |= (((psTime->tm_hour - 12) << HIB_CALLD0_HR_S) |
01515                          HIB_CAL0_AMPM);
01516         }
01517         else
01518         {
01519             //
01520             // All other times are normal and AM.
01521             //
01522             ui32Time |= (psTime->tm_hour << HIB_CALLD0_HR_S);
01523         }
01524     }
01525 
01526     //
01527     // Create the date in the correct register format.
01528     //
01529     if(ui32Reg == HIB_CAL0)
01530     {
01531         //
01532         // We must add 1 to the month, since the time structure lists
01533         // the month from 0 to 11 and the HIB lists it from 1 to 12.
01534         //
01535         ui32Date = ((psTime->tm_mday << HIB_CAL1_DOM_S) |
01536                     ((psTime->tm_mon + 1) << HIB_CAL1_MON_S) |
01537                     (psTime->tm_wday << HIB_CAL1_DOW_S) |
01538                     ((psTime->tm_year - 100) << HIB_CAL1_YEAR_S));
01539     }
01540     else
01541     {
01542         //
01543         // Wday, month and year are not included in the match
01544         // Functionality.
01545         //
01546         if(psTime->tm_mday == 0xFF)
01547         {
01548             //
01549             // program 0 to match every day
01550             //
01551             ui32Date = 0 << HIB_CAL1_DOM_M;
01552         }
01553         else
01554         {
01555             ui32Date = (psTime->tm_mday << HIB_CAL1_DOM_S);
01556         }
01557     }
01558 
01559     //
01560     // Load register requires unlock.
01561     //
01562     if(ui32Reg == HIB_CAL0)
01563     {
01564         //
01565         // Unlock the hibernate counter load registers.
01566         //
01567         HWREG(HIB_LOCK) = HIB_LOCK_HIBLOCK_KEY;
01568         _HibernateWriteComplete();
01569     }
01570 
01571     //
01572     // Set the requested time and date.
01573     //
01574     if(ui32Reg == HIB_CAL0)
01575     {
01576         HWREG(HIB_CALLD0) = ui32Time;
01577         _HibernateWriteComplete();
01578         HWREG(HIB_CALLD1) = ui32Date;
01579         _HibernateWriteComplete();
01580     }
01581     else
01582     {
01583         HWREG(HIB_CALM0) = ui32Time;
01584         _HibernateWriteComplete();
01585         HWREG(HIB_CALM1) = ui32Date;
01586         _HibernateWriteComplete();
01587     }
01588 
01589     //
01590     // Load register requires unlock.
01591     //
01592     if(ui32Reg == HIB_CAL0)
01593     {
01594         //
01595         // Lock the hibernate counter load registers.
01596         //
01597         HWREG(HIB_LOCK) = 0;
01598         _HibernateWriteComplete();
01599     }
01600 }

01632 //*****************************************************************************
01633 //
01658 //
01659 //*****************************************************************************
01660 int
01661 HibernateCalendarGet(struct tm *psTime)
01662 {
01663     uint32_t ui32Date, ui32Time;
01664 
01665     ASSERT(HWREG(HIB_CALCTL) & HIB_CALCTL_CALEN);
01666 
01667     //
01668     // Wait for the value to be valid, this should never be more than a few
01669     // loops and should never hang.
01670     //
01671     do
01672     {
01673         ui32Date = HWREG(HIB_CAL1);
01674     }
01675     while((ui32Date & HIB_CAL1_VALID) == 0);
01676 
01677     //
01678     // Wait for the value to be valid, this should never be more than a few
01679     // loops and should never hang.
01680     //
01681     do
01682     {
01683         ui32Time = HWREG(HIB_CAL0);
01684     }
01685     while((ui32Time & HIB_CAL0_VALID) == 0);
01686 
01687     //
01688     // The date changed after reading the time so fail this call and let the
01689     // application call again since it knows how int32_t to wait until another
01690     // second passes.
01691     //
01692     if(ui32Date != HWREG(HIB_CAL1))
01693     {
01694         return(-1);
01695     }
01696 
01697     //
01698     // Populate the date and time fields in the psTime structure.
01699     // We must subtract 1 from the month, since the time structure lists
01700     // the month from 0 to 11 and the HIB lists it from 1 to 12.
01701     //
01702     psTime->tm_min = (ui32Time & HIB_CAL0_MIN_M) >> HIB_CAL0_MIN_S;
01703     psTime->tm_sec = (ui32Time & HIB_CAL0_SEC_M) >> HIB_CAL0_SEC_S;
01704     psTime->tm_mon = (((ui32Date & HIB_CAL1_MON_M) >> HIB_CAL1_MON_S) - 1);
01705     psTime->tm_mday = (ui32Date & HIB_CAL1_DOM_M) >> HIB_CAL1_DOM_S;
01706     psTime->tm_wday = (ui32Date & HIB_CAL1_DOW_M) >> HIB_CAL1_DOW_S;
01707     psTime->tm_year = ((ui32Date & HIB_CAL1_YEAR_M) >> HIB_CAL1_YEAR_S) + 100;
01708     psTime->tm_hour = (ui32Time & HIB_CAL0_HR_M) >> HIB_CAL0_HR_S;
01709 
01710     //
01711     // Fix up the hour in the non-24-hour mode and the time is in PM.
01712     //
01713     if(((HWREG(HIB_CALCTL) & HIB_CALCTL_CAL24) == 0) &&
01714        (ui32Time & HIB_CAL0_AMPM))
01715     {
01716         psTime->tm_hour += 12;
01717     }
01718 
01719     return(0);
01720 }
