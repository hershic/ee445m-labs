00075 int32_t
00076 sine(uint32_t ui32Angle)
00077 {
00078     uint32_t ui32Idx;
00079 
00080     //
00081     // Add 0.5 to the angle.  Since only the upper 9 bits are used to compute
00082     // the sine value, adding one to the tenth bit is 0.5 from the point of
00083     // view of the sine table.
00084     //
00085     ui32Angle += 0x00400000;
00086 
00087     //
00088     // Get the index into the sine table from bits 30:23.
00089     //
00090     ui32Idx = (ui32Angle >> 23) & 255;
00091 
00092     //
00093     // If bit 30 is set, the angle is between 90 and 180 or 270 and 360.  In
00094     // these cases, the sine value is decreasing from one instead of increasing
00095     // from zero.  The indexing into the table needs to be reversed.
00096     //
00097     if(ui32Angle & 0x40000000)
00098     {
00099         ui32Idx = 256 - ui32Idx;
00100     }
00101 
00102     //
00103     // Get the value of the sine.
00104     //
00105     ui32Idx = g_pui16FixedSineTable[ui32Idx];
00106 
00107     //
00108     // If bit 31 is set, the angle is between 180 and 360.  In this case, the
00109     // sine value is negative; otherwise it is positive.
00110     //
00111     if(ui32Angle & 0x80000000)
00112     {
00113         return(0 - ui32Idx);
00114     }
00115     else
00116     {
00117         return(ui32Idx);
00118     }
00119 }
