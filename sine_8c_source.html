75 int32_t
76 sine(uint32_t ui32Angle)
77 {
78  uint32_t ui32Idx;
79 
80  //
81  // Add 0.5 to the angle. Since only the upper 9 bits are used to compute
82  // the sine value, adding one to the tenth bit is 0.5 from the point of
83  // view of the sine table.
84  //
85  ui32Angle += 0x00400000;
86 
87  //
88  // Get the index into the sine table from bits 30:23.
89  //
90  ui32Idx = (ui32Angle >> 23) & 255;
91 
92  //
93  // If bit 30 is set, the angle is between 90 and 180 or 270 and 360. In
94  // these cases, the sine value is decreasing from one instead of increasing
95  // from zero. The indexing into the table needs to be reversed.
96  //
97  if(ui32Angle & 0x40000000)
98  {
99  ui32Idx = 256 - ui32Idx;
100  }
101 
102  //
103  // Get the value of the sine.
104  //
105  ui32Idx = g_pui16FixedSineTable[ui32Idx];
106 
107  //
108  // If bit 31 is set, the angle is between 180 and 360. In this case, the
109  // sine value is negative; otherwise it is positive.
110  //
111  if(ui32Angle & 0x80000000)
112  {
113  return(0 - ui32Idx);
114  }
115  else
116  {
117  return(ui32Idx);
118  }
119 }
