00591 void
00592 SHAMD5ResultRead(uint32_t ui32Base, uint32_t *pui32Dest)
00593 {
00594     uint32_t ui32Idx, ui32Count;
00595 
00596     //
00597     // Check the arguments.
00598     //
00599     ASSERT(ui32Base == SHAMD5_BASE);
00600 
00601     //
00602     // Determine the number of bytes in the result, based on the hash type.
00603     //
00604     switch(HWREG(ui32Base + SHAMD5_O_MODE) & SHAMD5_MODE_ALGO_M)
00605     {
00606         //
00607         // The MD5 hash is being used.
00608         //
00609         case SHAMD5_MODE_ALGO_MD5:
00610         {
00611             //
00612             // There are 16 bytes in the MD5 hash.
00613             //
00614             ui32Count = 16;
00615 
00616             //
00617             // Done.
00618             //
00619             break;
00620         }
00621 
00622         //
00623         // The SHA-1 hash is being used.
00624         //
00625         case SHAMD5_MODE_ALGO_SHA1:
00626         {
00627             //
00628             // There are 20 bytes in the SHA-1 hash.
00629             //
00630             ui32Count = 20;
00631 
00632             //
00633             // Done.
00634             //
00635             break;
00636         }
00637 
00638         //
00639         // The SHA-224 hash is being used.
00640         //
00641         case SHAMD5_MODE_ALGO_SHA224:
00642         {
00643             //
00644             // There are 28 bytes in the SHA-224 hash.
00645             //
00646             ui32Count = 28;
00647 
00648             //
00649             // Done.
00650             //
00651             break;
00652         }
00653 
00654         //
00655         // The SHA-256 hash is being used.
00656         //
00657         case SHAMD5_MODE_ALGO_SHA256:
00658         {
00659             //
00660             // There are 32 bytes in the SHA-256 hash.
00661             //
00662             ui32Count = 32;
00663 
00664             //
00665             // Done.
00666             //
00667             break;
00668         }
00669 
00670         //
00671         // The hash type is not recognized.
00672         //
00673         default:
00674         {
00675             //
00676             // Return without reading a result since the hardware appears to be
00677             // misconfigured.
00678             //
00679             return;
00680         }
00681     }
00682 
00683     //
00684     // Read the hash result.
00685     //
00686     for(ui32Idx = 0; ui32Idx < ui32Count; ui32Idx += 4)
00687     {
00688         *pui32Dest++ = HWREG(ui32Base + SHAMD5_O_IDIGEST_A + ui32Idx);
00689     }
00690 }

00711 static void
00712 _SHAMD5DataWriteMultiple(uint32_t ui32Base, uint32_t *pui32DataSrc,
00713                          uint32_t ui32DataLength)
00714 {
00715     uint32_t ui32Idx, ui32Count;
00716 
00717     //
00718     // Check the arguments.
00719     //
00720     ASSERT(ui32Base == SHAMD5_BASE);
00721 
00722     //
00723     // Calculate the number of blocks of data.
00724     //
00725     ui32Count = ui32DataLength / 64;
00726 
00727     //
00728     // Loop through all the blocks and write them into the data registers
00729     // making sure to block additional operations until we can write the
00730     // next 16 words.
00731     //
00732     for(ui32Idx = 0; ui32Idx < ui32Count; ui32Idx++)
00733     {
00734         //
00735         // Write the block of data.
00736         //
00737         SHAMD5DataWrite(ui32Base, pui32DataSrc);
00738 
00739         //
00740         // Increment the pointer to next block of data.
00741         //
00742         pui32DataSrc += 16;
00743     }
00744 
00745     //
00746     // Calculate the remaining bytes of data that don't make up a full block.
00747     //
00748     ui32Count = ui32DataLength % 64;
00749 
00750     //
00751     // If there are bytes that do not make up a whole block, then
00752     // write them separately.
00753     //
00754     if(ui32Count)
00755     {
00756         //
00757         // Wait until the engine has finished processing the previous block.
00758         //
00759         while((HWREG(ui32Base + SHAMD5_O_IRQSTATUS) &
00760                SHAMD5_INT_INPUT_READY) == 0)
00761         {
00762         }
00763 
00764         //
00765         // Loop through the remaining words.
00766         //
00767         for(ui32Idx = 0; ui32Idx < ui32Count; ui32Idx += 4)
00768         {
00769             //
00770             // Write the word into the data register.
00771             //
00772             HWREG(ui32Base + SHAMD5_O_DATA_0_IN + ui32Idx) = *pui32DataSrc++;
00773         }
00774     }
00775 }
