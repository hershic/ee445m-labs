00604 {
00605     tDMAControlTable *psCtl;
00606 
00607     //
00608     // Check the arguments.
00609     //

00709 void
00710 uDMAChannelTransferSet(uint32_t ui32ChannelStructIndex, uint32_t ui32Mode,
00711                        void *pvSrcAddr, void *pvDstAddr,
00712                        uint32_t ui32TransferSize)
00713 {
00714     tDMAControlTable *psControlTable;
00715     uint32_t ui32Control;
00716     uint32_t ui32Inc;
00717     uint32_t ui32BufferBytes;
00718 
00719     //
00720     // Check the arguments.
00721     //
00722     ASSERT((ui32ChannelStructIndex & 0xffff) < 64);
00723     ASSERT(HWREG(UDMA_CTLBASE) != 0);
00724     ASSERT(ui32Mode <= UDMA_MODE_PER_SCATTER_GATHER);
00725     ASSERT((uint32_t)pvSrcAddr >= 0x20000000);
00726     ASSERT((uint32_t)pvDstAddr >= 0x20000000);
00727     ASSERT((ui32TransferSize != 0) && (ui32TransferSize <= 1024));
00728 
00729     //
00730     // In case a channel selector macro (like UDMA_CH0_USB0EP1RX) was
00731     // passed as the ui32ChannelStructIndex parameter, extract just the channel
00732     // index from this parameter.
00733     //
00734     ui32ChannelStructIndex &= 0x3f;
00735 
00736     //
00737     // Get the base address of the control table.
00738     //
00739     psControlTable = (tDMAControlTable *)HWREG(UDMA_CTLBASE);
00740 
00741     //
00742     // Get the current control word value and mask off the mode and size
00743     // fields.
00744     //
00745     ui32Control = (psControlTable[ui32ChannelStructIndex].ui32Control &
00746                    ~(UDMA_CHCTL_XFERSIZE_M | UDMA_CHCTL_XFERMODE_M));
00747 
00748     //
00749     // Adjust the mode if the alt control structure is selected.
00750     //
00751     if(ui32ChannelStructIndex & UDMA_ALT_SELECT)
00752     {
00753         if((ui32Mode == UDMA_MODE_MEM_SCATTER_GATHER) ||
00754            (ui32Mode == UDMA_MODE_PER_SCATTER_GATHER))
00755         {
00756             ui32Mode |= UDMA_MODE_ALT_SELECT;
00757         }
00758     }
00759 
00760     //
00761     // Set the transfer size and mode in the control word (but don't write the
00762     // control word yet as it could kick off a transfer).
00763     //
00764     ui32Control |= ui32Mode | ((ui32TransferSize - 1) << 4);
00765 
00766     //
00767     // Get the address increment value for the source, from the control word.
00768     //
00769     ui32Inc = (ui32Control & UDMA_CHCTL_SRCINC_M);
00770 
00771     //
00772     // Compute the ending source address of the transfer.  If the source
00773     // increment is set to none, then the ending address is the same as the
00774     // beginning.
00775     //
00776     if(ui32Inc != UDMA_SRC_INC_NONE)
00777     {
00778         ui32Inc = ui32Inc >> 26;
00779         ui32BufferBytes = ui32TransferSize << ui32Inc;
00780         pvSrcAddr = (void *)((uint32_t)pvSrcAddr + ui32BufferBytes - 1);
00781     }
00782 
00783     //
00784     // Load the source ending address into the control block.
00785     //
00786     psControlTable[ui32ChannelStructIndex].pvSrcEndAddr = pvSrcAddr;
00787 
00788     //
00789     // Get the address increment value for the destination, from the control
00790     // word.
00791     //
00792     ui32Inc = ui32Control & UDMA_CHCTL_DSTINC_M;
00793 
00794     //
00795     // Compute the ending destination address of the transfer.  If the
00796     // destination increment is set to none, then the ending address is the
00797     // same as the beginning.
00798     //
00799     if(ui32Inc != UDMA_DST_INC_NONE)
00800     {
00801         //
00802         // There is a special case if this is setting up a scatter-gather
00803         // transfer.  The destination pointer must point to the end of
00804         // the alternate structure for this channel instead of calculating
00805         // the end of the buffer in the normal way.
00806         //
00807         if((ui32Mode == UDMA_MODE_MEM_SCATTER_GATHER) ||
00808            (ui32Mode == UDMA_MODE_PER_SCATTER_GATHER))
00809         {
00810             pvDstAddr =
00811                 (void *)&psControlTable[ui32ChannelStructIndex |
00812                                         UDMA_ALT_SELECT].ui32Spare;
00813         }
00814         //
00815         // Not a scatter-gather transfer, calculate end pointer normally.
00816         //
00817         else
00818         {
00819             ui32Inc = ui32Inc >> 30;
00820             ui32BufferBytes = ui32TransferSize << ui32Inc;
00821             pvDstAddr = (void *)((uint32_t)pvDstAddr + ui32BufferBytes - 1);
00822         }
00823     }
00824 
00825     //
00826     // Load the destination ending address into the control block.
00827     //
00828     psControlTable[ui32ChannelStructIndex].pvDstEndAddr = pvDstAddr;
00829 
00830     //
00831     // Write the new control word value.
00832     //
00833     psControlTable[ui32ChannelStructIndex].ui32Control = ui32Control;
00834 }

00860 void
00861 uDMAChannelScatterGatherSet(uint32_t ui32ChannelNum, uint32_t ui32TaskCount,
00862                             void *pvTaskList, uint32_t ui32IsPeriphSG)
00863 {
00864     tDMAControlTable *psControlTable;
00865     tDMAControlTable *psTaskTable;
00866 
00867     //
00868     // Check the parameters
00869     //
00870     ASSERT((ui32ChannelNum & 0xffff) < 32);
00871     ASSERT(HWREG(UDMA_CTLBASE) != 0);
00872     ASSERT(pvTaskList != 0);
00873     ASSERT(ui32TaskCount <= 1024);
00874     ASSERT(ui32TaskCount != 0);
00875 
00876     //
00877     // In case a channel selector macro (like UDMA_CH0_USB0EP1RX) was
00878     // passed as the ui32ChannelNum parameter, extract just the channel number
00879     // from this parameter.
00880     //
00881     ui32ChannelNum &= 0x1f;
00882 
00883     //
00884     // Get the base address of the control table.
00885     //
00886     psControlTable = (tDMAControlTable *)HWREG(UDMA_CTLBASE);
00887 
00888     //
00889     // Get a handy pointer to the task list
00890     //
00891     psTaskTable = (tDMAControlTable *)pvTaskList;
00892 
00893     //
00894     // Compute the ending address for the source pointer.  This address is the
00895     // last element of the last task in the task table
00896     //
00897     psControlTable[ui32ChannelNum].pvSrcEndAddr =
00898         &psTaskTable[ui32TaskCount - 1].ui32Spare;
00899 
00900     //
00901     // Compute the ending address for the destination pointer.  This address
00902     // is the end of the alternate structure for this channel.
00903     //
00904     psControlTable[ui32ChannelNum].pvDstEndAddr =
00905         &psControlTable[ui32ChannelNum | UDMA_ALT_SELECT].ui32Spare;
00906 
00907     //
00908     // Compute the control word.  Most configurable items are fixed for
00909     // scatter-gather.  Item and increment sizes are all 32-bit and arb
00910     // size must be 4.  The count is the number of items in the task list
00911     // times 4 (4 words per task).
00912     //
00913     psControlTable[ui32ChannelNum].ui32Control =
00914         (UDMA_CHCTL_DSTINC_32 | UDMA_CHCTL_DSTSIZE_32 |
00915          UDMA_CHCTL_SRCINC_32 | UDMA_CHCTL_SRCSIZE_32 |
00916          UDMA_CHCTL_ARBSIZE_4 |
00917          (((ui32TaskCount * 4) - 1) << UDMA_CHCTL_XFERSIZE_S) |
00918          (ui32IsPeriphSG ? UDMA_CHCTL_XFERMODE_PER_SG :
00919           UDMA_CHCTL_XFERMODE_MEM_SG));
00920 
00921     //
00922     // Scatter-gather operations can leave the alt bit set.  So if doing
00923     // back to back scatter-gather transfers, the second attempt may not
00924     // work correctly because the alt bit is set.  Therefore, clear the
00925     // alt bit here to ensure that it is always cleared before a new SG
00926     // transfer is started.
00927     //
00928     HWREG(UDMA_ALTCLR) = 1 << ui32ChannelNum;
00929 }

00948 uDMAChannelSizeGet(uint32_t ui32ChannelStructIndex)
00949 {
00950     tDMAControlTable *psControlTable;
00951     uint32_t ui32Control;
00952 
00953     //
00954     // Check the arguments.
00955     //
00956     ASSERT((ui32ChannelStructIndex & 0xffff) < 64);
00957     ASSERT(HWREG(UDMA_CTLBASE) != 0);
00958 
00959     //
00960     // In case a channel selector macro (like UDMA_CH0_USB0EP1RX) was
00961     // passed as the ui32ChannelStructIndex parameter, extract just the channel
00962     // index from this parameter.
00963     //
00964     ui32ChannelStructIndex &= 0x3f;
00965 
00966     //
00967     // Get the base address of the control table.
00968     //
00969     psControlTable = (tDMAControlTable *)HWREG(UDMA_CTLBASE);
00970 
00971     //
00972     // Get the current control word value and mask off all but the size field
00973     // and the mode field.
00974     //
00975     ui32Control = (psControlTable[ui32ChannelStructIndex].ui32Control &
00976                    (UDMA_CHCTL_XFERSIZE_M | UDMA_CHCTL_XFERMODE_M));
00977 
00978     //
00979     // If the size field and mode field are 0 then the transfer is finished
00980     // and there are no more items to transfer
00981     //
00982     if(ui32Control == 0)
00983     {
00984         return(0);
00985     }
00986 
00987     //
00988     // Otherwise, if either the size field or more field is non-zero, then
00989     // not all the items have been transferred.
00990     //
00991     else
00992     {
00993         //
00994         // Shift the size field and add one, then return to user.
00995         //
00996         return((ui32Control >> 4) + 1);
00997     }
00998 }
