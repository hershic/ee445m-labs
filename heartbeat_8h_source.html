188  case GPIO_PORTB_BASE: SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOB); break;

193  case GPIO_PORTG_BASE: SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOG); break;
194  case GPIO_PORTH_BASE: SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOH); break;
195  case GPIO_PORTJ_BASE: SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOJ); break;
196  }
197 
198  /* Enable the GPIO pins for \HEART_ANCILLARY_MUSCLE. */
199  GPIOPinTypeGPIOOutput(base, pin);
200

202  HEART_MODAL_METADATA[os_running_thread_id()].base = base;
