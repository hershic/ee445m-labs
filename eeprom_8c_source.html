00093 static void
00094 _EEPROMSectorMaskSet(uint32_t ui32Address)
00095 {
00096     uint32_t ui32Mask;
00097 
00098     //
00099     // Determine which page contains the passed EEPROM address.  The 2KB EEPROM
00100     // is implemented in 16KB of flash with each 1KB sector of flash holding
00101     // values for 32 consecutive EEPROM words (or 128 bytes).
00102     //
00103     ui32Mask = ~(1 << (ui32Address >> 7));
00104 
00105     SysCtlDelay(10);
00106     HWREG(0x400FD0FC) = 3;
00107     SysCtlDelay(10);
00108     HWREG(0x400AE2C0) = ui32Mask;
00109     SysCtlDelay(10);
00110     HWREG(0x400FD0FC) = 0;
00111     SysCtlDelay(10);

00175 uint32_t
00176 EEPROMInit(void)
00177 {
00178     uint32_t ui32Status;
00179 
00180     //
00181     // Insert a small delay (6 cycles + call overhead) to guard against the
00182     // possibility that this function is called immediately after the EEPROM
00183     // peripheral is enabled.  Without this delay, there is a slight chance
00184     // that the first EEPROM register read will fault if you are using a
00185     // compiler with a ridiculously good optimizer!
00186     //
00187     SysCtlDelay(2);
00188 
00189     //
00190     // Make sure the EEPROM has finished any ongoing processing.
00191     //
00192     _EEPROMWaitForDone();
00193 
00194     //
00195     // Read the EESUPP register to see if any errors have been reported.
00196     //
00197     ui32Status = HWREG(EEPROM_EESUPP);
00198 
00199     //
00200     // Did an error of some sort occur during initialization?
00201     //
00202     if(ui32Status & (EEPROM_EESUPP_PRETRY | EEPROM_EESUPP_ERETRY))
00203     {
00204         return(EEPROM_INIT_ERROR);
00205     }
00206 
00207     //
00208     // Perform a second EEPROM reset.
00209     //
00210     SysCtlPeripheralReset(SYSCTL_PERIPH_EEPROM0);
00211 
00212     //
00213     // Wait for the EEPROM to complete its reset processing once again.
00214     //
00215     SysCtlDelay(2);
00216     _EEPROMWaitForDone();
00217 
00218     //
00219     // Read EESUPP once again to determine if any error occurred.
00220     //
00221     ui32Status = HWREG(EEPROM_EESUPP);
00222 
00223     //
00224     // Was an error reported following the second reset?
00225     //
00226     if(ui32Status & (EEPROM_EESUPP_PRETRY | EEPROM_EESUPP_ERETRY))
00227     {
00228         return(EEPROM_INIT_ERROR);
00229     }
00230 
00231     //
00232     // The EEPROM does not indicate that any error occurred.
00233     //
00234     return(EEPROM_INIT_OK);
00235 }

00306 void
00307 EEPROMRead(uint32_t *pui32Data, uint32_t ui32Address, uint32_t ui32Count)
00308 {
00309     //
00310     // Check parameters in a debug build.
00311     //
00312     ASSERT(pui32Data);
00313     ASSERT(ui32Address < SIZE_FROM_EESIZE(HWREG(EEPROM_EESIZE)));
00314     ASSERT((ui32Address + ui32Count) <=
00315            SIZE_FROM_EESIZE(HWREG(EEPROM_EESIZE)));
00316     ASSERT((ui32Address & 3) == 0);
00317     ASSERT((ui32Count & 3) == 0);
00318 
00319     //
00320     // Set the block and offset appropriately to read the first word.
00321     //
00322     HWREG(EEPROM_EEBLOCK) = EEPROMBlockFromAddr(ui32Address);
00323     HWREG(EEPROM_EEOFFSET) = OFFSET_FROM_ADDR(ui32Address);
00324 
00325     //
00326     // Convert the byte count to a word count.
00327     //
00328     ui32Count /= 4;
00329 
00330     //
00331     // Read each word in turn.
00332     //
00333     while(ui32Count)
00334     {
00335         //
00336         // Read the next word through the autoincrementing register.
00337         //
00338         *pui32Data = HWREG(EEPROM_EERDWRINC);
00339 
00340         //
00341         // Move on to the next word.
00342         //
00343         pui32Data++;
00344         ui32Count--;
00345 
00346         //
00347         // Do we need to move to the next block?  This is the case if the
00348         // offset register has just wrapped back to 0.  Note that we only
00349         // write the block register if we have more data to read.  If this
00350         // register is written, the hardware expects a read or write operation
00351         // next.  If a mass erase is requested instead, the mass erase will
00352         // fail.
00353         //
00354         if(ui32Count && (HWREG(EEPROM_EEOFFSET) == 0))
00355         {
00356             HWREG(EEPROM_EEBLOCK) += 1;
00357         }
00358     }
00359 }

00380 uint32_t
00381 EEPROMProgram(uint32_t *pui32Data, uint32_t ui32Address, uint32_t ui32Count)
00382 {
00383     uint32_t ui32Status;
00384 
00385     //
00386     // Check parameters in a debug build.
00387     //
00388     ASSERT(pui32Data);
00389     ASSERT(ui32Address < SIZE_FROM_EESIZE(HWREG(EEPROM_EESIZE)));
00390     ASSERT((ui32Address + ui32Count) <=
00391            SIZE_FROM_EESIZE(HWREG(EEPROM_EESIZE)));
00392     ASSERT((ui32Address & 3) == 0);
00393     ASSERT((ui32Count & 3) == 0);
00394 
00395     //
00396     // Make sure the EEPROM is idle before we start.
00397     //
00398     do
00399     {
00400         //
00401         // Read the status.
00402         //
00403         ui32Status = HWREG(EEPROM_EEDONE);
00404     }
00405     while(ui32Status & EEPROM_EEDONE_WORKING);
00406 
00407     //
00408     // Set the block and offset appropriately to program the first word.
00409     //
00410     HWREG(EEPROM_EEBLOCK) = EEPROMBlockFromAddr(ui32Address);
00411     HWREG(EEPROM_EEOFFSET) = OFFSET_FROM_ADDR(ui32Address);
00412 
00413     //
00414     // Convert the byte count to a word count.
00415     //
00416     ui32Count /= 4;
00417 
00418     //
00419     // Write each word in turn.
00420     //
00421     while(ui32Count)
00422     {
00423         //
00424         // This is a workaround for a silicon problem on Blizzard rev A.  We
00425         // need to do this before every word write to ensure that we don't
00426         // have problems in multi-word writes that span multiple flash sectors.
00427         //
00428         if(CLASS_IS_TM4C123 && REVISION_IS_A0)
00429         {
00430             _EEPROMSectorMaskSet(ui32Address);
00431         }
00432 
00433         //
00434         // Write the next word through the autoincrementing register.
00435         //
00436         HWREG(EEPROM_EERDWRINC) = *pui32Data;
00437 
00438         //
00439         // Wait a few cycles.  In some cases, the WRBUSY bit is not set
00440         // immediately and this prevents us from dropping through the polling
00441         // loop before the bit is set.
00442         //
00443         SysCtlDelay(10);
00444 
00445         //
00446         // Wait for the write to complete.
00447         //
00448         do
00449         {
00450             //
00451             // Read the status.
00452             //
00453             ui32Status = HWREG(EEPROM_EEDONE);
00454         }
00455         while(ui32Status & EEPROM_EEDONE_WORKING);
00456 
00457         //
00458         // Make sure we completed the write without errors.  Note that we
00459         // must check this per-word because write permission can be set per
00460         // block resulting in only a section of the write not being performed.
00461         //
00462         if(ui32Status & EEPROM_EEDONE_NOPERM)
00463         {
00464             //
00465             // An error was reported that would prevent the values from
00466             // being written correctly.
00467             //
00468             if(CLASS_IS_TM4C123 && REVISION_IS_A0)
00469             {
00470                 _EEPROMSectorMaskClear();
00471             }
00472             return(ui32Status);
00473         }
00474 
00475         //
00476         // Move on to the next word.
00477         //
00478         pui32Data++;
00479         ui32Count--;
00480 
00481         //
00482         // Do we need to move to the next block?  This is the case if the
00483         // offset register has just wrapped back to 0.  Note that we only
00484         // write the block register if we have more data to read.  If this
00485         // register is written, the hardware expects a read or write operation
00486         // next.  If a mass erase is requested instead, the mass erase will
00487         // fail.
00488         //
00489         if(ui32Count && (HWREG(EEPROM_EEOFFSET) == 0))
00490         {
00491             HWREG(EEPROM_EEBLOCK) += 1;
00492         }
00493     }
00494 
00495     //
00496     // Clear the sector protection bits to prevent possible problems when
00497     // programming the main flash array later.
00498     //
00499     if(CLASS_IS_TM4C123 && REVISION_IS_A0)
00500     {
00501         _EEPROMSectorMaskClear();
00502     }
00503 
00504     //
00505     // Return the current status to the caller.
00506     //
00507     return(HWREG(EEPROM_EEDONE));
00508 }

00901                   uint32_t ui32Count)

00904     // Check parameters in a debug build.
00905     //
00906     ASSERT(pui32Password);
00907     ASSERT(ui32Block < BLOCKS_FROM_EESIZE(HWREG(EEPROM_EESIZE)));
00908     ASSERT(ui32Count <= 3);

00912     //
00913     HWREG(EEPROM_EEBLOCK) = ui32Block;

00927 
00928     //
00929     // Write the supplied password to unlock the block.
00930     //
00931     while(ui32Count)
00932     {
00933         HWREG(EEPROM_EEUNLOCK) = *pui32Password--;
00934         ui32Count--;
00935     }
00936 
00937     //
00938     // Let the caller know if their password worked.
00939     //
00940     return(HWREG(EEPROM_EEUNLOCK));
00941 }

01062 uint32_t
01063 EEPROMIntStatus(bool bMasked)
01064 {
01065     if(bMasked)
01066     {
01067         //
01068         // If asked for the masked interrupt status, we check to see if the
01069         // relevant interrupt is pending in the flash controller then return
01070         // the appropriate EEPROM flag if it is.
01071         //
01072         return((HWREG(FLASH_FCMISC) & FLASH_FCMISC_EMISC) ?
01073                EEPROM_INT_PROGRAM : 0);
01074     }
01075     else
01076     {
01077         //
01078         // If asked for the unmasked interrupt status, infer that an interrupt
01079         // is pending if the WORKING bit of the EEDONE register is clear.  The
01080         // actual interrupt fires on the high to low transition of this bit
01081         // but we don't have access to an unmasked interrupt status for the
01082         // EEPROM because it's handled via the flash controller so we have to
01083         // make do with this instead.
01084         //
01085         return((HWREG(EEPROM_EEDONE) & EEPROM_EEDONE_WORKING) ?
01086                0 : EEPROM_INT_PROGRAM);
01087     }
01088 }
