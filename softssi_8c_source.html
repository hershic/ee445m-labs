00090 SoftSSIConfigSet(tSoftSSI *psSSI, uint8_t ui8Protocol,
00091                  uint8_t ui8Bits)
00092 {
00093     //
00094     // See if a GPIO pin has been set for Fss.
00095     //
00096     if(psSSI->ui32FssGPIO != 0)
00097     {
00098         //
00099         // Configure the Fss pin.
00100         //
00101         MAP_GPIOPinTypeGPIOOutput(psSSI->ui32FssGPIO & 0xfffff000,
00102                                   (psSSI->ui32FssGPIO & 0x00000fff) >> 2);
00103 
00104         //
00105         // Set the Fss pin high.
00106         //
00107         HWREG(psSSI->ui32FssGPIO) = 255;
00108     }
00109 
00110     //
00111     // Configure the Clk pin.
00112     //
00113     MAP_GPIOPinTypeGPIOOutput(psSSI->ui32ClkGPIO & 0xfffff000,
00114                               (psSSI->ui32ClkGPIO & 0x00000fff) >> 2);
00115 
00116     //
00117     // Set the Clk pin high or low based on the configured clock polarity.
00118     //
00119     if((ui8Protocol & SOFTSSI_FLAG_SPO) == 0)
00120     {
00121         HWREG(psSSI->ui32ClkGPIO) = 0;
00122     }
00123     else
00124     {
00125         HWREG(psSSI->ui32ClkGPIO) = 255;
00126     }
00127 
00128     //
00129     // Configure the Tx pin and set it low.
00130     //
00131     MAP_GPIOPinTypeGPIOOutput(psSSI->ui32TxGPIO & 0xfffff000,
00132                               (psSSI->ui32TxGPIO & 0x00000fff) >> 2);
00133     HWREG(psSSI->ui32TxGPIO) = 0;
00134 
00135     //
00136     // See if a GPIO pin has been set for Rx.
00137     //
00138     if(psSSI->ui32RxGPIO != 0)
00139     {
00140         //
00141         // Configure the Rx pin.
00142         //
00143         MAP_GPIOPinTypeGPIOInput(psSSI->ui32RxGPIO & 0xfffff000,
00144                                  (psSSI->ui32RxGPIO & 0x00000fff) >> 2);
00145     }
00146 
00147     //
00148     // Make sure that the transmit and receive FIFOs are empty.
00149     //
00150     psSSI->ui16TxBufferRead = 0;
00151     psSSI->ui16TxBufferWrite = 0;
00152     psSSI->ui16RxBufferRead = 0;
00153     psSSI->ui16RxBufferWrite = 0;
00154 
00155     //
00156     // Save the frame protocol.
00157     //
00158     psSSI->ui8Flags = ui8Protocol;
00159 
00160     //
00161     // Save the number of data bits.
00162     //
00163     psSSI->ui8Bits = ui8Bits;
00164 
00165     //
00166     // Since the FIFOs are empty, the transmit FIFO "interrupt" is asserted.
00167     //
00168     psSSI->ui8IntStatus = SOFTSSI_TXFF;
00169 
00170     //
00171     // Reset the idle counter.
00172     //
00173     psSSI->ui8IdleCount = 0;
00174 
00175     //
00176     // Disable the SoftSSI module.
00177     //
00178     psSSI->ui8Flags &= ~(SOFTSSI_FLAG_ENABLE);
00179 
00180     //
00181     // Start the SoftSSI state machine in the idle state.
00182     //
00183     psSSI->ui8State = SOFTSSI_STATE_IDLE;
00184 }

00198 static void
00199 SoftSSITxInt(tSoftSSI *psSSI)
00200 {
00201     uint16_t ui16Temp;
00202 
00203     //
00204     // Determine the number of words left in the transmit FIFO.
00205     //
00206     if(psSSI->ui16TxBufferRead > psSSI->ui16TxBufferWrite)
00207     {
00208         ui16Temp = (psSSI->ui16TxBufferLen + psSSI->ui16TxBufferWrite -
00209                     psSSI->ui16TxBufferRead);
00210     }
00211     else
00212     {
00213         ui16Temp = psSSI->ui16TxBufferWrite - psSSI->ui16TxBufferRead;
00214     }
00215 
00216     //
00217     // If the transmit FIFO is now half full or less, generate a transmit FIFO
00218     // "interrupt".  Otherwise, clear the transmit FIFO "interrupt".
00219     //
00220     if(ui16Temp <= (psSSI->ui16TxBufferLen / 2))
00221     {
00222         psSSI->ui8IntStatus |= SOFTSSI_TXFF;
00223     }
00224     else
00225     {
00226         psSSI->ui8IntStatus &= ~(SOFTSSI_TXFF);
00227     }

00292 SoftSSITimerTick(tSoftSSI *psSSI)
00293 {
00294     uint16_t ui16Temp;
00295 
00296     //
00297     // Determine the current state of the state machine.
00298     //
00299     switch(psSSI->ui8State)
00300     {
00301         //
00302         // The state machine is idle.
00303         //
00304         case SOFTSSI_STATE_IDLE:
00305         {
00306             //
00307             // See if the SoftSSI module is enabled and there is data in the
00308             // transmit FIFO.
00309             //
00310             if(((psSSI->ui8Flags & SOFTSSI_FLAG_ENABLE) != 0) &&
00311                (psSSI->ui16TxBufferRead != psSSI->ui16TxBufferWrite))
00312             {
00313                 //
00314                 // Assert the Fss signal if it is configured.
00315                 //
00316                 if(psSSI->ui32FssGPIO != 0)
00317                 {
00318                     HWREG(psSSI->ui32FssGPIO) = 0;
00319                 }
00320 
00321                 //
00322                 // Move to the start state.
00323                 //
00324                 psSSI->ui8State = SOFTSSI_STATE_START;
00325             }
00326 
00327             //
00328             // Otherwise, see if there is data in the receive FIFO.
00329             //
00330             else if((psSSI->ui16RxBufferRead != psSSI->ui16RxBufferWrite) &&
00331                     (psSSI->ui8IdleCount != 64))
00332             {
00333                 //
00334                 // Increment the idle counter.
00335                 //
00336                 psSSI->ui8IdleCount++;
00337 
00338                 //
00339                 // See if the idle counter has become large enough to trigger
00340                 // a timeout "interrupt".
00341                 //
00342                 if(psSSI->ui8IdleCount == 64)
00343                 {
00344                     //
00345                     // Trigger the receive timeout "interrupt".
00346                     //
00347                     psSSI->ui8IntStatus |= SOFTSSI_RXTO;
00348                 }
00349             }
00350 
00351             //
00352             // This state has been handled.
00353             //
00354             break;
00355         }
00356 
00357         //
00358         // The start machine is in the transfer start state.
00359         //
00360         case SOFTSSI_STATE_START:
00361         {
00362             //
00363             // Get the next word to transfer from the transmit FIFO.
00364             //
00365             psSSI->ui16TxData =
00366                 (psSSI->pui16TxBuffer[psSSI->ui16TxBufferRead] <<
00367                  (16 - psSSI->ui8Bits));
00368 
00369             //
00370             // Initialize the receive buffer to zero.
00371             //
00372             psSSI->ui16RxData = 0;
00373 
00374             //
00375             // Initialize the count of bits tranferred.
00376             //
00377             psSSI->ui8CurrentBit = 0;
00378 
00379             //
00380             // Write the first bit of the transmit word to the Tx pin.
00381             //
00382             HWREG(psSSI->ui32TxGPIO) =
00383                 (psSSI->ui16TxData & 0x8000) ? 255 : 0;
00384 
00385             //
00386             // Shift to the next bit of the transmit word.
00387             //
00388             psSSI->ui16TxData <<= 1;
00389 
00390             //
00391             // If in SPI mode 1 or 3, then the Clk signal needs to be toggled.
00392             //
00393             if((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) != 0)
00394             {
00395                 HWREG(psSSI->ui32ClkGPIO) ^= 255;
00396             }
00397 
00398             //
00399             // Move to the data input state.
00400             //
00401             psSSI->ui8State = SOFTSSI_STATE_IN;
00402 
00403             //
00404             // This state has been handled.
00405             //
00406             break;
00407         }
00408 
00409         //
00410         // The state machine is in the data input state.
00411         //
00412         case SOFTSSI_STATE_IN:
00413         {
00414             //
00415             // Read the next bit from the Rx signal if it is configured.
00416             //
00417             if(psSSI->ui32RxGPIO != 0)
00418             {
00419                 psSSI->ui16RxData = ((psSSI->ui16RxData << 1) |
00420                                      (HWREG(psSSI->ui32RxGPIO) ? 1 : 0));
00421             }
00422 
00423             //
00424             // Toggle the Clk signal.
00425             //
00426             HWREG(psSSI->ui32ClkGPIO) ^= 255;
00427 
00428             //
00429             // Increment the number of bits transferred.
00430             //
00431             psSSI->ui8CurrentBit++;
00432 
00433             //
00434             // See if the entire word has been transferred.
00435             //
00436             if(psSSI->ui8CurrentBit != psSSI->ui8Bits)
00437             {
00438                 //
00439                 // There are more bits to transfer, so move to the data output
00440                 // state.
00441                 //
00442                 psSSI->ui8State = SOFTSSI_STATE_OUT;
00443             }
00444             else
00445             {
00446                 //
00447                 // Increment the transmit read pointer, removing the word that
00448                 // was just transferred from the transmit FIFO.
00449                 //
00450                 psSSI->ui16TxBufferRead++;
00451                 if(psSSI->ui16TxBufferRead == psSSI->ui16TxBufferLen)
00452                 {
00453                     psSSI->ui16TxBufferRead = 0;
00454                 }
00455 
00456                 //
00457                 // See if a transmit FIFO "interrupt" needs to be asserted.
00458                 //
00459                 SoftSSITxInt(psSSI);
00460 
00461                 //
00462                 // Determine the new value for the receive FIFO write pointer.
00463                 //
00464                 ui16Temp = psSSI->ui16RxBufferWrite + 1;
00465                 if(ui16Temp >= psSSI->ui16RxBufferLen)
00466                 {
00467                     ui16Temp = 0;
00468                 }
00469 
00470                 //
00471                 // See if there is space in the receive FIFO for the word that
00472                 // was just received.
00473                 //
00474                 if(ui16Temp == psSSI->ui16RxBufferRead)
00475                 {
00476                     //
00477                     // The receive FIFO is full, so generate a receive FIFO
00478                     // overrun "interrupt".
00479                     //
00480                     psSSI->ui8IntStatus |= SOFTSSI_RXOR;
00481                 }
00482                 else
00483                 {
00484                     //
00485                     // Store the new word into the receive FIFO.
00486                     //
00487                     psSSI->pui16RxBuffer[psSSI->ui16RxBufferWrite] =
00488                         psSSI->ui16RxData;
00489 
00490                     //
00491                     // Save the new receive FIFO write pointer.
00492                     //
00493                     psSSI->ui16RxBufferWrite = ui16Temp;
00494 
00495                     //
00496                     // See if a receive FIFO "interrupt" needs to be asserted.
00497                     //
00498                     SoftSSIRxInt(psSSI);
00499                 }
00500 
00501                 //
00502                 // See if the next word should be transmitted immediately.
00503                 // This will occur when there is data in the transmit FIFO, the
00504                 // SoftSSI module is enabled, and the SoftSSI module is in SPI
00505                 // mode 1 or 3.
00506                 //
00507                 if(((psSSI->ui8Flags & SOFTSSI_FLAG_ENABLE) != 0) &&
00508                    ((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) != 0) &&
00509                    (psSSI->ui16TxBufferRead != psSSI->ui16TxBufferWrite))
00510                 {
00511                     //
00512                     // Get the next word to transfer from the transmit FIFO.
00513                     //
00514                     psSSI->ui16TxData =
00515                         (psSSI->pui16TxBuffer[psSSI->ui16TxBufferRead] <<
00516                          (16 - psSSI->ui8Bits));
00517 
00518                     //
00519                     // Initialize the receive buffer to zero.
00520                     //
00521                     psSSI->ui16RxData = 0;
00522 
00523                     //
00524                     // Initialize the count of bits tranferred.
00525                     //
00526                     psSSI->ui8CurrentBit = 0;
00527 
00528                     //
00529                     // Move to the data output state.
00530                     //
00531                     psSSI->ui8State = SOFTSSI_STATE_OUT;
00532                 }
00533                 else
00534                 {
00535                     //
00536                     // The next word should not be transmitted immediately, so
00537                     // move to the first step of the stop state.
00538                     //
00539                     psSSI->ui8State = SOFTSSI_STATE_STOP1;
00540                 }
00541             }
00542 
00543             //
00544             // This state has been handled.
00545             //
00546             break;
00547         }
00548 
00549         //
00550         // The state machine is in the data output state.
00551         //
00552         case SOFTSSI_STATE_OUT:
00553         {
00554             //
00555             // Write the next bit of the transmit word to the Tx pin.
00556             //
00557             HWREG(psSSI->ui32TxGPIO) = (psSSI->ui16TxData & 0x8000) ? 255 : 0;
00558 
00559             //
00560             // Toggle the Clk signal.
00561             //
00562             HWREG(psSSI->ui32ClkGPIO) ^= 255;
00563 
00564             //
00565             // Shift to the next bit of the transmit word.
00566             //
00567             psSSI->ui16TxData <<= 1;
00568 
00569             //
00570             // Move to the data input state.
00571             //
00572             psSSI->ui8State = SOFTSSI_STATE_IN;
00573 
00574             //
00575             // This state has been handled.
00576             //
00577             break;
00578         }
00579 
00580         //
00581         // The state machine is in the first step of the stop state.
00582         //
00583         case SOFTSSI_STATE_STOP1:
00584         {
00585             //
00586             // Set the Tx pin low.
00587             //
00588             HWREG(psSSI->ui32TxGPIO) = 0;
00589 
00590             //
00591             // If in SPI mode 1 or 3, then the Clk signal needs to be toggled.
00592             //
00593             if((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) == 0)
00594             {
00595                 HWREG(psSSI->ui32ClkGPIO) ^= 255;
00596             }
00597 
00598             //
00599             // Move to the second step of the stop state.
00600             //
00601             psSSI->ui8State = SOFTSSI_STATE_STOP2;
00602 
00603             //
00604             // This state has been handled.
00605             //
00606             break;
00607         }
00608 
00609         //
00610         // The state machine is in the second step of the stop state.
00611         //
00612         case SOFTSSI_STATE_STOP2:
00613         {
00614             //
00615             // Deassert the Fss signal if it is configured.
00616             //
00617             if(psSSI->ui32FssGPIO != 0)
00618             {
00619                 HWREG(psSSI->ui32FssGPIO) = 255;
00620             }
00621 
00622             //
00623             // Move to the idle state.
00624             //
00625             psSSI->ui8State = SOFTSSI_STATE_IDLE;
00626 
00627             //

00629             //
00630             psSSI->ui8IdleCount = 0;
00631

00633             // See if the end of transfer "interrupt" should be generated.
00634             //
00635             if(psSSI->ui16TxBufferRead == psSSI->ui16TxBufferWrite)

00637                 psSSI->ui8IntStatus |= SOFTSSI_TXEOT;
00638             }
00639 
00640             //
00641             // This state has been handled.
00642             //
00643             break;

00645     }

00652     //

00654           (psSSI->pfnIntCallback != 0))
00655     {
00656         //
00657         // Call the callback function.
00658         //
00659         psSSI->pfnIntCallback();

00899         ui16Temp = 0;
