89 void
90 SoftSSIConfigSet(tSoftSSI *psSSI, uint8_t ui8Protocol,
91  uint8_t ui8Bits)
92 {
93  //
94  // See if a GPIO pin has been set for Fss.
95  //
96  if(psSSI->ui32FssGPIO != 0)
97  {
98  //
99  // Configure the Fss pin.
100  //
101  MAP_GPIOPinTypeGPIOOutput(psSSI->ui32FssGPIO & 0xfffff000,
102  (psSSI->ui32FssGPIO & 0x00000fff) >> 2);
103 
104  //
105  // Set the Fss pin high.
106  //
107  HWREG(psSSI->ui32FssGPIO) = 255;
108  }
109 
110  //
111  // Configure the Clk pin.
112  //
113  MAP_GPIOPinTypeGPIOOutput(psSSI->ui32ClkGPIO & 0xfffff000,
114  (psSSI->ui32ClkGPIO & 0x00000fff) >> 2);
115 
116  //
117  // Set the Clk pin high or low based on the configured clock polarity.
118  //
119  if((ui8Protocol & SOFTSSI_FLAG_SPO) == 0)
120  {
121  HWREG(psSSI->ui32ClkGPIO) = 0;
122  }
123  else
124  {
125  HWREG(psSSI->ui32ClkGPIO) = 255;
126  }
127 
128  //
129  // Configure the Tx pin and set it low.
130  //
131  MAP_GPIOPinTypeGPIOOutput(psSSI->ui32TxGPIO & 0xfffff000,
132  (psSSI->ui32TxGPIO & 0x00000fff) >> 2);
133  HWREG(psSSI->ui32TxGPIO) = 0;
134 
135  //
136  // See if a GPIO pin has been set for Rx.
137  //
138  if(psSSI->ui32RxGPIO != 0)
139  {
140  //
141  // Configure the Rx pin.
142  //
143  MAP_GPIOPinTypeGPIOInput(psSSI->ui32RxGPIO & 0xfffff000,
144  (psSSI->ui32RxGPIO & 0x00000fff) >> 2);
145  }
146 
147  //
148  // Make sure that the transmit and receive FIFOs are empty.
149  //
150  psSSI->ui16TxBufferRead = 0;
151  psSSI->ui16TxBufferWrite = 0;
152  psSSI->ui16RxBufferRead = 0;
153  psSSI->ui16RxBufferWrite = 0;
154 
155  //
156  // Save the frame protocol.
157  //
158  psSSI->ui8Flags = ui8Protocol;
159 
160  //
161  // Save the number of data bits.
162  //
163  psSSI->ui8Bits = ui8Bits;
164 
165  //
166  // Since the FIFOs are empty, the transmit FIFO "interrupt" is asserted.
167  //
168  psSSI->ui8IntStatus = SOFTSSI_TXFF;
169 
170  //
171  // Reset the idle counter.
172  //
173  psSSI->ui8IdleCount = 0;
174 
175  //
176  // Disable the SoftSSI module.
177  //
178  psSSI->ui8Flags &= ~(SOFTSSI_FLAG_ENABLE);
179 
180  //
181  // Start the SoftSSI state machine in the idle state.
182  //
183  psSSI->ui8State = SOFTSSI_STATE_IDLE;
184 }

198 static void
199 SoftSSITxInt(tSoftSSI *psSSI)
200 {
201  uint16_t ui16Temp;
202 
203  //
204  // Determine the number of words left in the transmit FIFO.
205  //
206  if(psSSI->ui16TxBufferRead > psSSI->ui16TxBufferWrite)
207  {
208  ui16Temp = (psSSI->ui16TxBufferLen + psSSI->ui16TxBufferWrite -
209  psSSI->ui16TxBufferRead);
210  }
211  else
212  {
213  ui16Temp = psSSI->ui16TxBufferWrite - psSSI->ui16TxBufferRead;
214  }
215 
216  //
217  // If the transmit FIFO is now half full or less, generate a transmit FIFO
218  // "interrupt". Otherwise, clear the transmit FIFO "interrupt".
219  //
220  if(ui16Temp <= (psSSI->ui16TxBufferLen / 2))
221  {
222  psSSI->ui8IntStatus |= SOFTSSI_TXFF;
223  }
224  else
225  {
226  psSSI->ui8IntStatus &= ~(SOFTSSI_TXFF);
227  }
228 }

242 static void
243 SoftSSIRxInt(tSoftSSI *psSSI)
244 {
245  uint16_t ui16Temp;
246 
247  //
248  // Determine the number of words in the receive FIFO.
249  //
250  if(psSSI->ui16RxBufferRead > psSSI->ui16RxBufferWrite)
251  {
252  ui16Temp = (psSSI->ui16RxBufferLen + psSSI->ui16RxBufferWrite -
253  psSSI->ui16RxBufferRead);
254  }
255  else
256  {
257  ui16Temp = psSSI->ui16RxBufferWrite - psSSI->ui16RxBufferRead;
258  }
259 
260  //
261  // If the receive FIFO is now half full or more, generate a receive FIFO
262  // "interrupt". Otherwise, clear the receive FIFO "interrupt".
263  //
264  if(ui16Temp >= (psSSI->ui16RxBufferLen / 2))
265  {
266  psSSI->ui8IntStatus |= SOFTSSI_RXFF;
267  }
268  else
269  {
270  psSSI->ui8IntStatus &= ~(SOFTSSI_RXFF);
271  }
272 }

291 void
292 SoftSSITimerTick(tSoftSSI *psSSI)
293 {
294  uint16_t ui16Temp;
295 
296  //
297  // Determine the current state of the state machine.
298  //
299  switch(psSSI->ui8State)
300  {
301  //
302  // The state machine is idle.
303  //
304  case SOFTSSI_STATE_IDLE:
305  {
306  //
307  // See if the SoftSSI module is enabled and there is data in the
308  // transmit FIFO.
309  //
310  if(((psSSI->ui8Flags & SOFTSSI_FLAG_ENABLE) != 0) &&
311  (psSSI->ui16TxBufferRead != psSSI->ui16TxBufferWrite))
312  {
313  //
314  // Assert the Fss signal if it is configured.
315  //
316  if(psSSI->ui32FssGPIO != 0)
317  {
318  HWREG(psSSI->ui32FssGPIO) = 0;
319  }
320 
321  //
322  // Move to the start state.
323  //
324  psSSI->ui8State = SOFTSSI_STATE_START;
325  }
326 
327  //
328  // Otherwise, see if there is data in the receive FIFO.
329  //
330  else if((psSSI->ui16RxBufferRead != psSSI->ui16RxBufferWrite) &&
331  (psSSI->ui8IdleCount != 64))
332  {
333  //
334  // Increment the idle counter.
335  //
336  psSSI->ui8IdleCount++;
337 
338  //
339  // See if the idle counter has become large enough to trigger
340  // a timeout "interrupt".
341  //
342  if(psSSI->ui8IdleCount == 64)
343  {
344  //
345  // Trigger the receive timeout "interrupt".
346  //
347  psSSI->ui8IntStatus |= SOFTSSI_RXTO;
348  }
349  }
350 
351  //
352  // This state has been handled.
353  //
354  break;
355  }
356 
357  //
358  // The start machine is in the transfer start state.
359  //
360  case SOFTSSI_STATE_START:
361  {
362  //
363  // Get the next word to transfer from the transmit FIFO.
364  //
365  psSSI->ui16TxData =
366  (psSSI->pui16TxBuffer[psSSI->ui16TxBufferRead] <<
367  (16 - psSSI->ui8Bits));
368 
369  //
370  // Initialize the receive buffer to zero.
371  //
372  psSSI->ui16RxData = 0;
373 
374  //
375  // Initialize the count of bits tranferred.
376  //
377  psSSI->ui8CurrentBit = 0;
378 
379  //
380  // Write the first bit of the transmit word to the Tx pin.
381  //
382  HWREG(psSSI->ui32TxGPIO) =
383  (psSSI->ui16TxData & 0x8000) ? 255 : 0;
384 
385  //
386  // Shift to the next bit of the transmit word.
387  //
388  psSSI->ui16TxData <<= 1;
389 
390  //
391  // If in SPI mode 1 or 3, then the Clk signal needs to be toggled.
392  //
393  if((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) != 0)
394  {
395  HWREG(psSSI->ui32ClkGPIO) ^= 255;
396  }
397 
398  //
399  // Move to the data input state.
400  //
401  psSSI->ui8State = SOFTSSI_STATE_IN;
402 
403  //
404  // This state has been handled.
405  //
406  break;
407  }
408 
409  //
410  // The state machine is in the data input state.
411  //
412  case SOFTSSI_STATE_IN:
413  {
414  //
415  // Read the next bit from the Rx signal if it is configured.
416  //
417  if(psSSI->ui32RxGPIO != 0)
418  {
419  psSSI->ui16RxData = ((psSSI->ui16RxData << 1) |
420  (HWREG(psSSI->ui32RxGPIO) ? 1 : 0));
421  }
422 
423  //
424  // Toggle the Clk signal.
425  //
426  HWREG(psSSI->ui32ClkGPIO) ^= 255;
427 
428  //
429  // Increment the number of bits transferred.
430  //
431  psSSI->ui8CurrentBit++;
432 
433  //
434  // See if the entire word has been transferred.
435  //
436  if(psSSI->ui8CurrentBit != psSSI->ui8Bits)
437  {
438  //
439  // There are more bits to transfer, so move to the data output
440  // state.
441  //
442  psSSI->ui8State = SOFTSSI_STATE_OUT;
443  }
444  else
445  {
446  //
447  // Increment the transmit read pointer, removing the word that
448  // was just transferred from the transmit FIFO.
449  //
450  psSSI->ui16TxBufferRead++;
451  if(psSSI->ui16TxBufferRead == psSSI->ui16TxBufferLen)
452  {
453  psSSI->ui16TxBufferRead = 0;
454  }
455 
456  //
457  // See if a transmit FIFO "interrupt" needs to be asserted.
458  //
459  SoftSSITxInt(psSSI);
460 
461  //
462  // Determine the new value for the receive FIFO write pointer.
463  //
464  ui16Temp = psSSI->ui16RxBufferWrite + 1;
465  if(ui16Temp >= psSSI->ui16RxBufferLen)
466  {
467  ui16Temp = 0;
468  }
469 
470  //
471  // See if there is space in the receive FIFO for the word that
472  // was just received.
473  //
474  if(ui16Temp == psSSI->ui16RxBufferRead)
475  {
476  //
477  // The receive FIFO is full, so generate a receive FIFO
478  // overrun "interrupt".
479  //
480  psSSI->ui8IntStatus |= SOFTSSI_RXOR;
481  }
482  else
483  {
484  //
485  // Store the new word into the receive FIFO.
486  //
487  psSSI->pui16RxBuffer[psSSI->ui16RxBufferWrite] =
488  psSSI->ui16RxData;
489 
490  //
491  // Save the new receive FIFO write pointer.
492  //
493  psSSI->ui16RxBufferWrite = ui16Temp;
494 
495  //
496  // See if a receive FIFO "interrupt" needs to be asserted.
497  //
498  SoftSSIRxInt(psSSI);
499  }
500 
501  //
502  // See if the next word should be transmitted immediately.
503  // This will occur when there is data in the transmit FIFO, the
504  // SoftSSI module is enabled, and the SoftSSI module is in SPI
505  // mode 1 or 3.
506  //
507  if(((psSSI->ui8Flags & SOFTSSI_FLAG_ENABLE) != 0) &&
508  ((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) != 0) &&
509  (psSSI->ui16TxBufferRead != psSSI->ui16TxBufferWrite))
510  {
511  //
512  // Get the next word to transfer from the transmit FIFO.
513  //
514  psSSI->ui16TxData =
515  (psSSI->pui16TxBuffer[psSSI->ui16TxBufferRead] <<
516  (16 - psSSI->ui8Bits));
517 
518  //
519  // Initialize the receive buffer to zero.
520  //
521  psSSI->ui16RxData = 0;
522 
523  //
524  // Initialize the count of bits tranferred.
525  //
526  psSSI->ui8CurrentBit = 0;
527 
528  //
529  // Move to the data output state.
530  //
531  psSSI->ui8State = SOFTSSI_STATE_OUT;
532  }
533  else
534  {
535  //
536  // The next word should not be transmitted immediately, so
537  // move to the first step of the stop state.
538  //
539  psSSI->ui8State = SOFTSSI_STATE_STOP1;
540  }
541  }
542 
543  //
544  // This state has been handled.
545  //
546  break;
547  }
548 
549  //
550  // The state machine is in the data output state.
551  //
552  case SOFTSSI_STATE_OUT:
553  {
554  //
555  // Write the next bit of the transmit word to the Tx pin.
556  //
557  HWREG(psSSI->ui32TxGPIO) = (psSSI->ui16TxData & 0x8000) ? 255 : 0;
558 
559  //
560  // Toggle the Clk signal.
561  //
562  HWREG(psSSI->ui32ClkGPIO) ^= 255;
563 
564  //
565  // Shift to the next bit of the transmit word.
566  //
567  psSSI->ui16TxData <<= 1;
568 
569  //
570  // Move to the data input state.
571  //
572  psSSI->ui8State = SOFTSSI_STATE_IN;
573 
574  //
575  // This state has been handled.
576  //
577  break;
578  }
579 
580  //
581  // The state machine is in the first step of the stop state.
582  //
583  case SOFTSSI_STATE_STOP1:
584  {
585  //
586  // Set the Tx pin low.
587  //
588  HWREG(psSSI->ui32TxGPIO) = 0;
589 
590  //
591  // If in SPI mode 1 or 3, then the Clk signal needs to be toggled.
592  //
593  if((psSSI->ui8Flags & SOFTSSI_FLAG_SPH) == 0)
594  {
595  HWREG(psSSI->ui32ClkGPIO) ^= 255;
596  }
597 
598  //
599  // Move to the second step of the stop state.
600  //
601  psSSI->ui8State = SOFTSSI_STATE_STOP2;
602 
603  //
604  // This state has been handled.
605  //
606  break;
607  }
608 
609  //
610  // The state machine is in the second step of the stop state.
611  //
612  case SOFTSSI_STATE_STOP2:
613  {
614  //
615  // Deassert the Fss signal if it is configured.
616  //
617  if(psSSI->ui32FssGPIO != 0)
618  {
619  HWREG(psSSI->ui32FssGPIO) = 255;
620  }
621 
622  //
623  // Move to the idle state.
624  //
625  psSSI->ui8State = SOFTSSI_STATE_IDLE;
626 
627  //
628  // Reset the idle counter.
629  //
630  psSSI->ui8IdleCount = 0;
631 
632  //
633  // See if the end of transfer "interrupt" should be generated.
634  //
635  if(psSSI->ui16TxBufferRead == psSSI->ui16TxBufferWrite)
636  {
637  psSSI->ui8IntStatus |= SOFTSSI_TXEOT;
638  }
639 
640  //
641  // This state has been handled.
642  //
643  break;
644  }
645  }
646 
647  //
648  // Call the "interrupt" callback while there are enabled "interrupts"
649  // asserted. By calling in a loop until the "interrupts" are no longer
650  // asserted, this mimics the behavior of a real hardware implementation of
651  // the SSI peripheral.
652  //
653  while(((psSSI->ui8IntStatus & psSSI->ui8IntMask) != 0) &&
654  (psSSI->pfnIntCallback != 0))
655  {
656  //
657  // Call the callback function.
658  //
659  psSSI->pfnIntCallback();
660  }
661 }

911  //

914  SoftSSITxInt(psSSI);

986 void
987 SoftSSIDataGet(tSoftSSI *psSSI, uint32_t *pui32Data)
988 {
989  //
990  // Wait until there is data to be read.
991  //
992  while(psSSI->ui16RxBufferRead ==
993  *(volatile uint16_t *)(&(psSSI->ui16RxBufferWrite)))
994  {
995  }
996 
997  //
998  // Read data from SoftSSI.
999  //
