00117 void
00118 FPUStackingEnable(void)
00119 {
00120     //
00121     // Enable automatic state preservation for the floating-point unit, and
00122     // disable lazy state preservation (meaning that the floating-point state
00123     // is always stacked when floating-point instructions are used).
00124     //
00125     HWREG(NVIC_FPCC) = (HWREG(NVIC_FPCC) & ~NVIC_FPCC_LSPEN) | NVIC_FPCC_ASPEN;
00126 }
