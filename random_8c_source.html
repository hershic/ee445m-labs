91 uint32_t
92 RandomSeed(void)
93 {
94  uint32_t ui32A, ui32B, ui32C, ui32D, ui32Temp, ui32Idx;
95 
96  //
97  // Initialize the digest.
98  //
99  ui32A = 0x67452301;
100  ui32B = 0xefcdab89;
101  ui32C = 0x98badcfe;
102  ui32D = 0x10325476;
103 
104  //
105  // Perform the first round of operations.
106  //
107 #define F(a, b, c, d, k, s) \
108  { \
109  ui32Temp = a + (d ^ (b & (c ^ d))) + g_pui32RandomEntropy[k]; \
110  a = (ui32Temp << s) | (ui32Temp >> (32 - s)); \
111  }
112  for(ui32Idx = 0; ui32Idx < 16; ui32Idx += 4)
113  {
114  F(ui32A, ui32B, ui32C, ui32D, ui32Idx + 0, 3);
115  F(ui32D, ui32A, ui32B, ui32C, ui32Idx + 1, 7);
116  F(ui32C, ui32D, ui32A, ui32B, ui32Idx + 2, 11);
117  F(ui32B, ui32C, ui32D, ui32A, ui32Idx + 3, 19);
118  }
119 
120  //
121  // Perform the second round of operations.
122  //
123 #define G(a, b, c, d, k, s) \
124  { \
125  ui32Temp = (a + ((b & c) | (b & d) | (c & d)) + \
126  g_pui32RandomEntropy[k] + 0x5a827999); \
127  a = (ui32Temp << s) | (ui32Temp >> (32 - s)); \
128  }
129  for(ui32Idx = 0; ui32Idx < 4; ui32Idx++)
130  {
131  G(ui32A, ui32B, ui32C, ui32D, ui32Idx + 0, 3);
132  G(ui32D, ui32A, ui32B, ui32C, ui32Idx + 4, 5);
133  G(ui32C, ui32D, ui32A, ui32B, ui32Idx + 8, 9);
134  G(ui32B, ui32C, ui32D, ui32A, ui32Idx + 12, 13);
135  }
136 
137  //
138  // Perform the third round of operations.
139  //
140 #define H(a, b, c, d, k, s) \
141  { \
142  ui32Temp = a + (b ^ c ^ d) + g_pui32RandomEntropy[k] + 0x6ed9eba1; \
143  a = (ui32Temp << s) | (ui32Temp >> (32 - s)); \
144  }
145  for(ui32Idx = 0; ui32Idx < 4; ui32Idx += 2)
146  {
147  H(ui32A, ui32B, ui32C, ui32D, ui32Idx + 0, 3);
148  H(ui32D, ui32A, ui32B, ui32C, ui32Idx + 8, 9);
149  H(ui32C, ui32D, ui32A, ui32B, ui32Idx + 4, 11);
150  H(ui32B, ui32C, ui32D, ui32A, ui32Idx + 12, 15);
151 
152  if(ui32Idx == 2)
153  {
154  ui32Idx -= 3;
155  }
156  }
157 
158  //
159  // Use the first word of the resulting digest as the random number seed.
160  //
161  return(ui32A + 0x67452301);
162 }
