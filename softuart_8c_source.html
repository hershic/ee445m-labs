195 void
196 SoftUARTConfigSet(tSoftUART *psUART, uint32_t ui32Config)
197 {
198  //
199  // See if a GPIO pin has been set for Tx.
200  //
201  if(psUART->ui32TxGPIO != 0)
202  {
203  //
204  // Configure the Tx pin.
205  //
206  MAP_GPIOPinTypeGPIOOutput(psUART->ui32TxGPIO & 0xfffff000,
207  (psUART->ui32TxGPIO & 0x00000fff) >> 2);
208 
209  //
210  // Set the Tx pin high.
211  //
212  HWREG(psUART->ui32TxGPIO) = 255;
213  }
214 
215  //
216  // See if a GPIO pin has been set for Rx.
217  //
218  if(psUART->ui32RxGPIOPort != 0)
219  {
220  //
221  // Configure the Rx pin.
222  //
223  MAP_GPIOPinTypeGPIOInput(psUART->ui32RxGPIOPort, psUART->ui8RxPin);
224 
225  //
226  // Set the Rx pin to generate an interrupt on the next falling edge.
227  //
228  MAP_GPIOIntTypeSet(psUART->ui32RxGPIOPort, psUART->ui8RxPin,
229  GPIO_FALLING_EDGE);
230 
231  //
232  // Enable the Rx pin interrupt.
233  //
234  GPIOIntClear(psUART->ui32RxGPIOPort, psUART->ui8RxPin);
235  GPIOIntEnable(psUART->ui32RxGPIOPort, psUART->ui8RxPin);
236  }
237 
238  //
239  // Make sure that the transmit and receive buffers are empty.
240  //
241  psUART->ui16TxBufferRead = 0;
242  psUART->ui16TxBufferWrite = 0;
243  psUART->ui16RxBufferRead = 0;
244  psUART->ui16RxBufferWrite = 0;
245 
246  //
247  // Save the data format.
248  //
249  psUART->ui16Config = ((psUART->ui16Config & SOFTUART_CONFIG_EXT_M) |
250  (ui32Config & SOFTUART_CONFIG_BASE_M));
251 
252  //
253  // Enable the SoftUART module.
254  //
255  psUART->ui8Flags |= SOFTUART_FLAG_ENABLE;
256 
257  //
258  // The next value to be written to the Tx pin is one since the SoftUART is
259  // idle.
260  //
261  psUART->ui8TxNext = 255;
262 
263  //
264  // Start the SoftUART state machines in the idle state.
265  //
266  psUART->ui8TxState = SOFTUART_TXSTATE_IDLE;
267  psUART->ui8RxState = SOFTUART_RXSTATE_IDLE;
268 }

287 void
288 SoftUARTTxTimerTick(tSoftUART *psUART)
289 {
290  uint32_t ui32Temp;
291 
292  //
293  // Write the next value to the Tx data line. This value was computed on
294  // the previous timer tick, which helps to reduce the jitter on the Tx
295  // edges (which is important since a UART connection does not contain a
296  // clock signal).
297  //
298  HWREG(psUART->ui32TxGPIO) = psUART->ui8TxNext;
299 
300  //
301  // Determine the current state of the state machine.
302  //
303  switch(psUART->ui8TxState)
304  {
305  //
306  // The state machine is idle.
307  //
308  case SOFTUART_TXSTATE_IDLE:
309  {
310  //
311  // See if the SoftUART module is enabled.
312  //
313  if(!(psUART->ui8Flags & SOFTUART_FLAG_ENABLE))
314  {
315  //
316  // The SoftUART module is not enabled, so do nothing and stay
317  // in the idle state.
318  //
319  break;
320  }
321 
322  //
323  // See if the break signal should be asserted.
324  //
325  else if(psUART->ui8Flags & SOFTUART_FLAG_TXBREAK)
326  {
327  //
328  // The data line should be driven low while in the break state.
329  //
330  psUART->ui8TxNext = 0;
331 
332  //
333  // Move to the break state.
334  //
335  psUART->ui8TxState = SOFTUART_TXSTATE_BREAK;
336  }
337 
338  //
339  // Otherwise, see if there is data in the transmit buffer.
340  //
341  else if(psUART->ui16TxBufferRead != psUART->ui16TxBufferWrite)
342  {
343  //
344  // The data line should be driven low to indicate a start bit.
345  //
346  psUART->ui8TxNext = 0;
347 
348  //
349  // Move to the start bit state.
350  //
351  psUART->ui8TxState = SOFTUART_TXSTATE_START;
352  }
353 
354  //
355  // This state has been handled.
356  //
357  break;
358  }
359 
360  //
361  // The state machine is in the start bit state.
362  //
363  case SOFTUART_TXSTATE_START:
364  {
365  //
366  // Get the next byte to be transmitted.
367  //
368  psUART->ui8TxData = psUART->pui8TxBuffer[psUART->ui16TxBufferRead];
369 
370  //
371  // The next value to be written to the data line is the LSB of the
372  // next data byte.
373  //
374  psUART->ui8TxNext = (psUART->ui8TxData & 1) ? 255 : 0;
375 
376  //
377  // Move to the data bit 0 state.
378  //
379  psUART->ui8TxState = SOFTUART_TXSTATE_DATA_0;
380 
381  //
382  // This state has been handled.
383  //
384  break;
385  }
386 
387  //
388  // In each of these states, a bit of the data byte must be output.
389  // This depends upon TXSTATE_DATA_n and TXSTATE_DATA_(n+1) being
390  // consecutively numbered.
391  //
392  case SOFTUART_TXSTATE_DATA_0:
393  case SOFTUART_TXSTATE_DATA_1:
394  case SOFTUART_TXSTATE_DATA_2:
395  case SOFTUART_TXSTATE_DATA_3:
396  {
397  //
398  // The next value to be written to the data line is the next bit of
399  // the data byte.
400  //
401  psUART->ui8TxNext =
402  (psUART->ui8TxData & (1 << psUART->ui8TxState)) ? 255 : 0;
403 
404  //
405  // Advance to the next state.
406  //
407  psUART->ui8TxState++;
408 
409  //
410  // This state has been handled.
411  //
412  break;
413  }
414 
415  //
416  // In each of these states, a bit of the data byte must be output.
417  // Additionally, based on the configuration of the SoftUART, this bit
418  // might be the last data bit of the data byte. This depends upon
419  // TXSTATE_DATA_n and TXSTATE_DATA_(n+1) being consecutively numbered.
420  //
421  case SOFTUART_TXSTATE_DATA_4:
422  case SOFTUART_TXSTATE_DATA_5:
423  case SOFTUART_TXSTATE_DATA_6:
424  case SOFTUART_TXSTATE_DATA_7:
425  {
426  //
427  // See if the bit that was just transferred is the last bit of the
428  // data byte (based on the configuration of the SoftUART).
429  //
430  if(((psUART->ui16Config & SOFTUART_CONFIG_WLEN_MASK) >>
431  SOFTUART_CONFIG_WLEN_S) ==
432  (psUART->ui8TxState - SOFTUART_TXSTATE_DATA_4))
433  {
434  //
435  // See if parity is enabled.
436  //
437  if((psUART->ui16Config & SOFTUART_CONFIG_PAR_MASK) !=
438  SOFTUART_CONFIG_PAR_NONE)
439  {
440  //
441  // See if the parity is set to one.
442  //
443  if((psUART->ui16Config & SOFTUART_CONFIG_PAR_MASK) ==
444  SOFTUART_CONFIG_PAR_ONE)
445  {
446  //
447  // The next value to be written to the data line is
448  // one.
449  //
450  psUART->ui8TxNext = 255;
451  }
452 
453  //
454  // Otherwise, see if the parity is set to zero.
455  //
456  else if((psUART->ui16Config & SOFTUART_CONFIG_PAR_MASK) ==
457  SOFTUART_CONFIG_PAR_ZERO)
458  {
459  //
460  // The next value to be written to the data line is
461  // zero.
462  //
463  psUART->ui8TxNext = 0;
464  }
465 
466  //
467  // Otherwise, there is either even or odd parity.
468  //
469  else
470  {
471  //
472  // Find the odd parity for the data byte.
473  //
474  psUART->ui8TxNext =
475  ((g_pui32ParityOdd[psUART->ui8TxData >> 5] &
476  (1 << (psUART->ui8TxData & 31))) ? 255 : 0);
477 
478  //
479  // If the parity is set to even, then invert the
480  // parity just computed (making it even parity).
481  //
482  if((psUART->ui16Config & SOFTUART_CONFIG_PAR_MASK) ==
483  SOFTUART_CONFIG_PAR_EVEN)
484  {
485  psUART->ui8TxNext ^= 255;
486  }
487  }
488 
489  //
490  // Advance to the parity state.
491  //
492  psUART->ui8TxState = SOFTUART_TXSTATE_PARITY;
493  }
494 
495  //
496  // Parity is not enabled.
497  //
498  else
499  {
500  //
501  // The next value to write to the data line is the stop
502  // bit.
503  //
504  psUART->ui8TxNext = 255;
505 
506  //
507  // See if there are one or two stop bits.
508  //
509  if((psUART->ui16Config & SOFTUART_CONFIG_STOP_MASK) ==
510  SOFTUART_CONFIG_STOP_TWO)
511  {
512  //
513  // Advance to the two stop bits state.
514  //
515  psUART->ui8TxState = SOFTUART_TXSTATE_STOP_0;
516  }
517  else
518  {
519  //
520  // Advance to the one stop bit state.
521  //
522  psUART->ui8TxState = SOFTUART_TXSTATE_STOP_1;
523  }
524  }
525  }
526 
527  //
528  // Otherwise, there are more data bits to transfer.
529  //
530  else
531  {
532  //
533  // The next value to be written to the data line is the next
534  // bit of the data byte.
535  //
536  psUART->ui8TxNext =
537  (psUART->ui8TxData & (1 << psUART->ui8TxState)) ? 255 : 0;
538 
539  //
540  // Advance to the next state.
541  //
542  psUART->ui8TxState++;
543  }
544 
545  //
546  // This state has been handled.
547  //
548  break;
549  }
550 
551  //
552  // The state machine is in the parity bit state.
553  //
554  case SOFTUART_TXSTATE_PARITY:
555  {
556  //
557  // The next value to write to the data line is the stop bit.
558  //
559  psUART->ui8TxNext = 255;
560 
561  //
562  // See if there are one or two stop bits.
563  //
564  if((psUART->ui16Config & SOFTUART_CONFIG_STOP_MASK) ==
565  SOFTUART_CONFIG_STOP_TWO)
566  {
567  //
568  // Advance to the two stop bits state.
569  //
570  psUART->ui8TxState = SOFTUART_TXSTATE_STOP_0;
571  }
572  else
573  {
574  //
575  // Advance to the one stop bit state.
576  //
577  psUART->ui8TxState = SOFTUART_TXSTATE_STOP_1;
578  }
579 
580  //
581  // This state has been handled.
582  //
583  break;
584  }
585 
586  //
587  // The state machine is in the two stop bits state.
588  //
589  case SOFTUART_TXSTATE_STOP_0:
590  {
591  //
592  // Advance to the one stop bit state.
593  //
594  psUART->ui8TxState = SOFTUART_TXSTATE_STOP_1;
595 
596  //
597  // This state has been handled.
598  //
599  break;
600  }
601 
602  //
603  // The state machine is in the one stop bit state.
604  //
605  case SOFTUART_TXSTATE_STOP_1:
606  {
607  //
608  // The data byte has been completely transferred, so advance the
609  // read pointer.
610  //
611  psUART->ui16TxBufferRead++;
612  if(psUART->ui16TxBufferRead == psUART->ui16TxBufferLen)
613  {
614  psUART->ui16TxBufferRead = 0;
615  }
616 
617  //
618  // Determine the number of characters in the transmit buffer.
619  //
620  if(psUART->ui16TxBufferRead > psUART->ui16TxBufferWrite)
621  {
622  ui32Temp = (psUART->ui16TxBufferLen -
623  (psUART->ui16TxBufferRead -
624  psUART->ui16TxBufferWrite));
625  }
626  else
627  {
628  ui32Temp = (psUART->ui16TxBufferWrite -
629  psUART->ui16TxBufferRead);
630  }
631 
632  //
633  // If the transmit buffer fullness just crossed the programmed
634  // level, generate a transmit "interrupt".
635  //
636  if(ui32Temp == psUART->ui16TxBufferLevel)
637  {
638  psUART->ui16IntStatus |= SOFTUART_INT_TX;
639  }
640 
641  //
642  // See if the SoftUART module is enabled.
643  //
644  if(!(psUART->ui8Flags & SOFTUART_FLAG_ENABLE))
645  {
646  //
647  // The SoftUART module is not enabled, so do advance to the
648  // idle state.
649  //
650  psUART->ui8TxState = SOFTUART_TXSTATE_IDLE;
651  }
652 
653  //
654  // See if the break signal should be asserted.
655  //
656  else if(psUART->ui8Flags & SOFTUART_FLAG_TXBREAK)
657  {
658  //
659  // The data line should be driven low while in the break state.
660  //
661  psUART->ui8TxNext = 0;
662 
663  //
664  // Move to the break state.
665  //
666  psUART->ui8TxState = SOFTUART_TXSTATE_BREAK;
667  }
668 
669  //
670  // Otherwise, see if there is data in the transmit buffer.
671  //
672  else if(psUART->ui16TxBufferRead != psUART->ui16TxBufferWrite)
673  {
674  //
675  // The data line should be driven low to indicate a start bit.
676  //
677  psUART->ui8TxNext = 0;
678 
679  //
680  // Move to the start bit state.
681  //
682  psUART->ui8TxState = SOFTUART_TXSTATE_START;
683  }
684 
685  //
686  // Otherwise, there is nothing to do.
687  //
688  else
689  {
690  //
691  // Assert the end of transmission "interrupt".
692  //
693  psUART->ui16IntStatus |= SOFTUART_INT_EOT;
694 
695  //
696  // Advance to the idle state.
697  //
698  psUART->ui8TxState = SOFTUART_TXSTATE_IDLE;
699  }
700 
701  //
702  // This state has been handled.
703  //
704  break;
705  }
706 
707  //
708  // The state machine is in the break state.
709  //
710  case SOFTUART_TXSTATE_BREAK:
711  {
712  //
713  // See if the break should be deasserted.
714  //
715  if(!(psUART->ui8Flags & SOFTUART_FLAG_ENABLE) ||
716  !(psUART->ui8Flags & SOFTUART_FLAG_TXBREAK))
717  {
718  //
719  // The data line should be driven high to indicate it is idle.
720  //
721  psUART->ui8TxNext = 255;
722 
723  //
724  // Advance to the idle state.
725  //
726  psUART->ui8TxState = SOFTUART_TXSTATE_IDLE;
727  }
728 
729  //
730  // This state has been handled.
731  //
732  break;
733  }
734  }
735

737  // Call the "interrupt" callback while there are enabled "interrupts"
738  // asserted. By calling in a loop until the "interrupts" are no longer
739  // asserted, this mimics the behavior of a real hardware implementation of
740  // the UART peripheral.
741  //
742  while(((psUART->ui16IntStatus & psUART->ui16IntMask) != 0) &&
743  (psUART->pfnIntCallback != 0))
744  {
745  //
746  // Call the callback function.
747  //
748  psUART->pfnIntCallback();
749  }

816 SoftUARTRxTick(tSoftUART *psUART, bool bEdgeInt)
817 {
818  uint32_t ui32PinState, ui32Temp, ui32Ret;
819 
820  //
821  // Read the current state of the Rx data line.
822  //
823  ui32PinState = MAP_GPIOPinRead(psUART->ui32RxGPIOPort, psUART->ui8RxPin);
824 
825  //
826  // The default return code inidicates that the receive timer does not need
827  // to be stopped.
828  //
829  ui32Ret = SOFTUART_RXTIMER_NOP;
830 
831  //
832  // See if this is an edge interrupt while delaying for the receive timeout
833  // interrupt.
834  //
835  if(bEdgeInt && (psUART->ui8RxState == SOFTUART_RXSTATE_DELAY))
836  {
837  //
838  // The receive timeout has been cancelled since the next character has
839  // started, so go to the idle state.
840  //
841  psUART->ui8RxState = SOFTUART_RXSTATE_IDLE;
842  }
843 
844  //
845  // Determine the current state of the state machine.
846  //
847  switch(psUART->ui8RxState)
848  {
849  //
850  // The state machine is idle.
851  //
852  case SOFTUART_RXSTATE_IDLE:
