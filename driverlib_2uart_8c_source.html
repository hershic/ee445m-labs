00137 _UARTIntNumberGet(uint32_t ui32Base)
00138 {
00139     uint_fast8_t ui8Idx, ui8Rows;
00140     const uint32_t (*ppui32UARTIntMap)[2];
00141 
00142     //
00143     // Default interrupt map.
00144     //
00145     ppui32UARTIntMap = g_ppui32UARTIntMap;
00146     ui8Rows = g_ui8UARTIntMapRows;
00147 
00148     if(CLASS_IS_TM4C129)
00149     {
00150         ppui32UARTIntMap = g_ppui32UARTIntMapSnowflake;
00151         ui8Rows = g_ui8UARTIntMapRowsSnowflake;
00152     }
00153 
00154     //
00155     // Loop through the table that maps UART base addresses to interrupt
00156     // numbers.
00157     //
00158     for(ui8Idx = 0; ui8Idx < ui8Rows; ui8Idx++)
00159     {
00160         //
00161         // See if this base address matches.
00162         //
00163         if(ppui32UARTIntMap[ui8Idx][0] == ui32Base)
00164         {
00165             //
00166             // Return the corresponding interrupt number.
00167             //
00168             return(ppui32UARTIntMap[ui8Idx][1]);
00169         }
00170     }
00171 
00172     //
00173     // The base address could not be found, so return an error.
00174     //
00175     return(0);

00367 void
00368 UARTConfigSetExpClk(uint32_t ui32Base, uint32_t ui32UARTClk,
00369                     uint32_t ui32Baud, uint32_t ui32Config)
00370 {
00371     uint32_t ui32Div;
00372 
00373     //
00374     // Check the arguments.
00375     //
00376     ASSERT(_UARTBaseValid(ui32Base));
00377     ASSERT(ui32Baud != 0);
00378     ASSERT(ui32UARTClk >= (ui32Baud * UART_CLK_DIVIDER));
00379 
00380     //
00381     // Stop the UART.
00382     //
00383     UARTDisable(ui32Base);
00384 
00385     //
00386     // Is the required baud rate greater than the maximum rate supported
00387     // without the use of high speed mode?
00388     //
00389     if((ui32Baud * 16) > ui32UARTClk)
00390     {
00391         //
00392         // Enable high speed mode.
00393         //
00394         HWREG(ui32Base + UART_O_CTL) |= UART_CTL_HSE;
00395 
00396         //
00397         // Half the supplied baud rate to compensate for enabling high speed
00398         // mode.  This allows the following code to be common to both cases.
00399         //
00400         ui32Baud /= 2;
00401     }
00402     else
00403     {
00404         //
00405         // Disable high speed mode.
00406         //
00407         HWREG(ui32Base + UART_O_CTL) &= ~(UART_CTL_HSE);
00408     }
00409 
00410     //
00411     // Compute the fractional baud rate divider.
00412     //
00413     ui32Div = (((ui32UARTClk * 8) / ui32Baud) + 1) / 2;
00414 
00415     //
00416     // Set the baud rate.
00417     //
00418     HWREG(ui32Base + UART_O_IBRD) = ui32Div / 64;
00419     HWREG(ui32Base + UART_O_FBRD) = ui32Div % 64;
00420 
00421     //
00422     // Set parity, data length, and number of stop bits.
00423     //
00424     HWREG(ui32Base + UART_O_LCRH) = ui32Config;
00425 
00426     //
00427     // Clear the flags register.
00428     //
00429     HWREG(ui32Base + UART_O_FR) = 0;
00430 
00431     //
00432     // Start the UART.
00433     //
00434     UARTEnable(ui32Base);
00435 }

00466 void
00467 UARTConfigGetExpClk(uint32_t ui32Base, uint32_t ui32UARTClk,
00468                     uint32_t *pui32Baud, uint32_t *pui32Config)
00469 {
00470     uint32_t ui32Int, ui32Frac;
00471 
00472     //
00473     // Check the arguments.
00474     //
00475     ASSERT(_UARTBaseValid(ui32Base));
00476 
00477     //
00478     // Compute the baud rate.
00479     //
00480     ui32Int = HWREG(ui32Base + UART_O_IBRD);
00481     ui32Frac = HWREG(ui32Base + UART_O_FBRD);
00482     *pui32Baud = (ui32UARTClk * 4) / ((64 * ui32Int) + ui32Frac);
00483 
00484     //
00485     // See if high speed mode enabled.
00486     //
00487     if(HWREG(ui32Base + UART_O_CTL) & UART_CTL_HSE)
00488     {
00489         //
00490         // High speed mode is enabled so the actual baud rate is actually
00491         // double what was just calculated.
00492         //
00493         *pui32Baud *= 2;
00494     }
00495 
00496     //
00497     // Get the parity, data length, and number of stop bits.
00498     //
00499     *pui32Config = (HWREG(ui32Base + UART_O_LCRH) &
00500                     (UART_LCRH_SPS | UART_LCRH_WLEN_M | UART_LCRH_STP2 |
00501                      UART_LCRH_EPS | UART_LCRH_PEN));
00502 }

00719 UARTSmartCardEnable(uint32_t ui32Base)
00720 {
00721     uint32_t ui32Val;
00722 
00723     //
00724     // Check the arguments.
00725     //
00726     ASSERT(_UARTBaseValid(ui32Base));
00727 
00728     //
00729     // Set 8-bit word length, even parity, 2 stop bits (note that although the
00730     // STP2 bit is ignored when in smartcard mode, this code lets the caller
00731     // read back the actual setting in use).
00732     //
00733     ui32Val = HWREG(ui32Base + UART_O_LCRH);
00734     ui32Val &= ~(UART_LCRH_SPS | UART_LCRH_EPS | UART_LCRH_PEN |
00735                  UART_LCRH_WLEN_M);
00736     ui32Val |= UART_LCRH_WLEN_8 | UART_LCRH_PEN | UART_LCRH_EPS |
00737                UART_LCRH_STP2;
00738     HWREG(ui32Base + UART_O_LCRH) = ui32Val;
00739 
00740     //
00741     // Enable SMART mode.
00742     //
00743     HWREG(ui32Base + UART_O_CTL) |= UART_CTL_SMART;
00744 }
