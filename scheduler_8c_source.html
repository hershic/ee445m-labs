00169 SchedulerTaskEnable(uint32_t ui32Index, bool bRunNow)
00170 {
00171     //
00172     // Is the task index passed valid?
00173     //
00174     if(ui32Index < g_ui32SchedulerNumTasks)
00175     {
00176         //
00177         // Yes - mark the task as active.
00178         //
00179         g_psSchedulerTable[ui32Index].bActive = true;
00180 
00181         //
00182         // Set the last call time to ensure that the function is called either
00183         // next time the scheduler is run or after the desired number of ticks
00184         // depending upon the value of the bRunNow parameter.
00185         //
00186         if(bRunNow)
00187         {
00188             //
00189             // Cause the task to run on the next call to SchedulerRun().
00190             //
00191             g_psSchedulerTable[ui32Index].ui32LastCall =
00192                 (g_ui32SchedulerTickCount -
00193                  g_psSchedulerTable[ui32Index].ui32FrequencyTicks);
00194         }
00195         else
00196         {
00197             //
00198             // Cause the task to run after one full time period.
00199             //
00200             g_psSchedulerTable[ui32Index].ui32LastCall =
00201                 g_ui32SchedulerTickCount;
