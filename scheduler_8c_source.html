113 void
114 SchedulerRun(void)
115 {
116  uint32_t ui32Loop;
117  tSchedulerTask *pi16Task;
118 
119  //
120  // Loop through each task in the task table.
121  //
122  for(ui32Loop = 0; ui32Loop < g_ui32SchedulerNumTasks; ui32Loop++)
123  {
124  //
125  // Get a pointer to the task information.
126  //
127  pi16Task = &g_psSchedulerTable[ui32Loop];
128 
129  //
130  // Is this task active and, if so, is it time to call it's function?
131  //
132  if(pi16Task->bActive &&
133  (SchedulerElapsedTicksGet(pi16Task->ui32LastCall) >=
134  pi16Task->ui32FrequencyTicks))
135  {
136  //
137  // Remember the timestamp at which we make the function call.
138  //
139  pi16Task->ui32LastCall = g_ui32SchedulerTickCount;
140 
141  //
142  // Call the task function, passing the provided parameter.
143  //
144  pi16Task->pfnFunction(pi16Task->pvParam);
145  }
146  }
147 }

168 void
169 SchedulerTaskEnable(uint32_t ui32Index, bool bRunNow)
170 {
171  //
172  // Is the task index passed valid?
173  //
174  if(ui32Index < g_ui32SchedulerNumTasks)
175  {
176  //
177  // Yes - mark the task as active.
178  //
179  g_psSchedulerTable[ui32Index].bActive = true;
180 
181  //
182  // Set the last call time to ensure that the function is called either
183  // next time the scheduler is run or after the desired number of ticks
184  // depending upon the value of the bRunNow parameter.
185  //
186  if(bRunNow)
187  {
188  //
189  // Cause the task to run on the next call to SchedulerRun().
190  //
191  g_psSchedulerTable[ui32Index].ui32LastCall =
192  (g_ui32SchedulerTickCount -
193  g_psSchedulerTable[ui32Index].ui32FrequencyTicks);
194  }
195  else
196  {
197  //
198  // Cause the task to run after one full time period.
199  //
200  g_psSchedulerTable[ui32Index].ui32LastCall =
201  g_ui32SchedulerTickCount;
202  }
203  }
204 }
