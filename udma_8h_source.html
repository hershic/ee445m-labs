00064 // fields are used by the uDMA controller and normally it is not necessary for
00065 // software to directly read or write fields in the table.

00161 #define uDMATaskStructEntry(ui32TransferCount,                                \
00162                             ui32ItemSize,                                     \
00163                             ui32SrcIncrement,                                 \
00164                             pvSrcAddr,                                        \
00165                             ui32DstIncrement,                                 \
00166                             pvDstAddr,                                        \
00167                             ui32ArbSize,                                      \
00168                             ui32Mode)                                         \
00169     {                                                                         \
00170         (((ui32SrcIncrement) == UDMA_SRC_INC_NONE) ? (void *)(pvSrcAddr) :    \
00171             ((void *)(&((uint8_t *)(pvSrcAddr))[((ui32TransferCount) <<       \
00172                                          ((ui32SrcIncrement) >> 26)) - 1]))), \
00173             (((ui32DstIncrement) == UDMA_DST_INC_NONE) ? (void *)(pvDstAddr) :\
00174             ((void *)(&((uint8_t *)(pvDstAddr))[((ui32TransferCount) <<       \
00175                                          ((ui32DstIncrement) >> 30)) - 1]))), \
00176         (ui32SrcIncrement) | (ui32DstIncrement) | (ui32ItemSize) |            \
00177             (ui32ArbSize) |                                                   \
00178         (((ui32TransferCount) - 1) << 4) |                                    \
00179         ((((ui32Mode) == UDMA_MODE_MEM_SCATTER_GATHER) ||                     \
00180           ((ui32Mode) == UDMA_MODE_PER_SCATTER_GATHER)) ?                     \
00181                 (ui32Mode) | UDMA_MODE_ALT_SELECT : (ui32Mode)), 0            \
00182     }
