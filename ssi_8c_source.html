00118 static uint32_t
00119 _SSIIntNumberGet(uint32_t ui32Base)
00120 {
00121     uint_fast8_t ui8Idx, ui8Rows;
00122     const uint32_t (*ppui32SSIIntMap)[2];
00123 
00124     //
00125     // Check the arguments.
00126     //
00127     ASSERT(_SSIBaseValid(ui32Base));
00128 
00129     ppui32SSIIntMap = g_ppui32SSIIntMap;
00130     ui8Rows = g_ui8SSIIntMapRows;
00131 
00132     if(CLASS_IS_TM4C129)
00133     {
00134         ppui32SSIIntMap = g_ppui32SSIIntMapSnowflake;
00135         ui8Rows = g_ui8SSIIntMapSnowflakeRows;
00136     }
00137 
00138     //
00139     // Loop through the table that maps SSI base addresses to interrupt
00140     // numbers.
00141     //
00142     for(ui8Idx = 0; ui8Idx < ui8Rows; ui8Idx++)
00143     {
00144         //
00145         // See if this base address matches.
00146         //
00147         if(ppui32SSIIntMap[ui8Idx][0] == ui32Base)
00148         {
00149             //
00150             // Return the corresponding interrupt number.
00151             //
00152             return(ppui32SSIIntMap[ui8Idx][1]);
00153         }
00154     }
00155 
00156     //
00157     // The base address could not be found, so return an error.
00158     //
00159     return(0);
00160 }

00220 void
00221 SSIConfigSetExpClk(uint32_t ui32Base, uint32_t ui32SSIClk,
00222                    uint32_t ui32Protocol, uint32_t ui32Mode,
00223                    uint32_t ui32BitRate, uint32_t ui32DataWidth)
00224 {
00225     uint32_t ui32MaxBitRate;
00226     uint32_t ui32RegVal;
00227     uint32_t ui32PreDiv;
00228     uint32_t ui32SCR;
00229     uint32_t ui32SPH_SPO;
00230 
00231     //
00232     // Check the arguments.
00233     //
00234     ASSERT(_SSIBaseValid(ui32Base));
00235     ASSERT((ui32Protocol == SSI_FRF_MOTO_MODE_0) ||
00236            (ui32Protocol == SSI_FRF_MOTO_MODE_1) ||
00237            (ui32Protocol == SSI_FRF_MOTO_MODE_2) ||
00238            (ui32Protocol == SSI_FRF_MOTO_MODE_3) ||
00239            (ui32Protocol == SSI_FRF_TI) ||
00240            (ui32Protocol == SSI_FRF_NMW));
00241     ASSERT((ui32Mode == SSI_MODE_MASTER) ||
00242            (ui32Mode == SSI_MODE_SLAVE) ||
00243            (ui32Mode == SSI_MODE_SLAVE_OD));
00244     ASSERT(((ui32Mode == SSI_MODE_MASTER) &&
00245             (ui32BitRate <= (ui32SSIClk / 2))) ||
00246            ((ui32Mode != SSI_MODE_MASTER) &&
00247             (ui32BitRate <= (ui32SSIClk / 12))));
00248     ASSERT((ui32SSIClk / ui32BitRate) <= (254 * 256));
00249     ASSERT((ui32DataWidth >= 4) && (ui32DataWidth <= 16));
00250 
00251     //
00252     // Set the mode.
00253     //
00254     ui32RegVal = (ui32Mode == SSI_MODE_SLAVE_OD) ? SSI_CR1_SOD : 0;
00255     ui32RegVal |= (ui32Mode == SSI_MODE_MASTER) ? 0 : SSI_CR1_MS;
00256     HWREG(ui32Base + SSI_O_CR1) = ui32RegVal;
00257 
00258     //
00259     // Set the clock predivider.
00260     //
00261     ui32MaxBitRate = ui32SSIClk / ui32BitRate;
00262     ui32PreDiv = 0;
00263     do
00264     {
00265         ui32PreDiv += 2;
00266         ui32SCR = (ui32MaxBitRate / ui32PreDiv) - 1;
00267     }
00268     while(ui32SCR > 255);
00269     HWREG(ui32Base + SSI_O_CPSR) = ui32PreDiv;
00270 
00271     //
00272     // Set protocol and clock rate.
00273     //
00274     ui32SPH_SPO = (ui32Protocol & 3) << 6;
00275     ui32Protocol &= SSI_CR0_FRF_M;
00276     ui32RegVal = (ui32SCR << 8) | ui32SPH_SPO | ui32Protocol |
00277                  (ui32DataWidth - 1);
00278     HWREG(ui32Base + SSI_O_CR0) = ui32RegVal;
00279 }
