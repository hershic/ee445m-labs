00130 static uint32_t
00131 _I2CIntNumberGet(uint32_t ui32Base)
00132 {
00133     int_fast8_t i8Idx, i8Rows;
00134     const uint32_t (*ppui32I2CIntMap)[2];
00135 
00136     //
00137     // Check the arguments.
00138     //
00139     ASSERT(_I2CBaseValid(ui32Base));
00140 
00141     ppui32I2CIntMap = g_ppui32I2CIntMap;
00142     i8Rows = g_i8I2CIntMapRows;
00143 
00144     if(CLASS_IS_TM4C129)
00145     {
00146         ppui32I2CIntMap = g_ppui32I2CIntMapSnowflake;
00147         i8Rows = g_i8I2CIntMapSnowflakeRows;
00148     }
00149 
00150     //
00151     // Loop through the table that maps I2C base addresses to interrupt
00152     // numbers.
00153     //
00154     for(i8Idx = 0; i8Idx < i8Rows; i8Idx++)
00155     {
00156         //
00157         // See if this base address matches.
00158         //
00159         if(ppui32I2CIntMap[i8Idx][0] == ui32Base)
00160         {
00161             //
00162             // Return the corresponding interrupt number.
00163             //
00164             return(ppui32I2CIntMap[i8Idx][1]);
00165         }
00166     }
00167 
00168     //
00169     // The base address could not be found, so return an error.
00170     //
00171     return(0);
00172 }

00201 void
00202 I2CMasterInitExpClk(uint32_t ui32Base, uint32_t ui32I2CClk,
00203                     bool bFast)
00204 {
00205     uint32_t ui32SCLFreq;
00206     uint32_t ui32TPR;
00207 
00208     //
00209     // Check the arguments.
00210     //
00211     ASSERT(_I2CBaseValid(ui32Base));
00212 
00213     //
00214     // Must enable the device before doing anything else.
00215     //
00216     I2CMasterEnable(ui32Base);
00217 
00218     //
00219     // Get the desired SCL speed.
00220     //
00221     if(bFast == true)
00222     {
00223         ui32SCLFreq = 400000;
00224     }
00225     else
00226     {
00227         ui32SCLFreq = 100000;
00228     }
00229 
00230     //
00231     // Compute the clock divider that achieves the fastest speed less than or
00232     // equal to the desired speed.  The numerator is biased to favor a larger
00233     // clock divider so that the resulting clock is always less than or equal
00234     // to the desired clock, never greater.
00235     //
00236     ui32TPR = ((ui32I2CClk + (2 * 10 * ui32SCLFreq) - 1) /
00237                (2 * 10 * ui32SCLFreq)) - 1;
00238     HWREG(ui32Base + I2C_O_MTPR) = ui32TPR;
00239 
00240     //
00241     // Check to see if this I2C peripheral is High-Speed enabled.  If yes, also
00242     // choose the fastest speed that is less than or equal to 3.4 Mbps.
00243     //
00244     if(HWREG(ui32Base + I2C_O_PP) & I2C_PP_HS)
00245     {
00246         ui32TPR = ((ui32I2CClk + (2 * 3 * 3400000) - 1) /
00247                    (2 * 3 * 3400000)) - 1;
00248         HWREG(ui32Base + I2C_O_MTPR) = I2C_MTPR_HS | ui32TPR;
00249     }
00250 }
