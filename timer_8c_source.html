00145 static uint32_t
00146 _TimerIntNumberGet(uint32_t ui32Base, uint32_t ui32Timer)
00147 {
00148     uint32_t ui32Int;
00149     uint_fast8_t ui8Idx, ui8Rows;
00150     const uint32_t (*ppui32SSIIntMap)[2];
00151 
00152     //
00153     // Default interrupt map.
00154     //
00155     ppui32SSIIntMap = g_ppui32TimerIntMap;
00156     ui8Rows = g_ui8TimerIntMapRows;
00157 
00158     if(CLASS_IS_TM4C129)
00159     {
00160         ppui32SSIIntMap = g_ppui32TimerIntMapSnowflake;
00161         ui8Rows = g_ui8TimerIntMapRowsSnowflake;
00162     }
00163 
00164     //
00165     // Loop through the table that maps timer base addresses to interrupt
00166     // numbers.
00167     //
00168     for(ui8Idx = 0; ui8Idx < ui8Rows; ui8Idx++)
00169     {
00170         //
00171         // See if this base address matches.
00172         //
00173         if(ppui32SSIIntMap[ui8Idx][0] == ui32Base)
00174         {
00175             ui32Int = ppui32SSIIntMap[ui8Idx][1];
00176 
00177             if(ui32Timer == TIMER_B)
00178             {
00179                 ui32Int += 1;
00180             }
00181 
00182             //
00183             // Return the corresponding interrupt number.
00184             //
00185             return(ui32Int);
00186         }
00187     }
00188 
00189     //
00190     // The base address could not be found, so return an error.
00191     //
00192     return(0);
00193 }

00350     // Check the arguments.

00358            ((ui32Config & 0xff000000) == TIMER_CFG_SPLIT_PAIR));

00361              ((ui32Config & 0x000000ff) == TIMER_CFG_A_ONE_SHOT_UP) ||
00362              ((ui32Config & 0x000000ff) == TIMER_CFG_A_PERIODIC) ||
00363              ((ui32Config & 0x000000ff) == TIMER_CFG_A_PERIODIC_UP) ||
00364              ((ui32Config & 0x000000ff) == TIMER_CFG_A_CAP_COUNT) ||
00365              ((ui32Config & 0x000000ff) == TIMER_CFG_A_CAP_TIME) ||
00366              ((ui32Config & 0x000000ff) == TIMER_CFG_A_PWM)) &&
00367             (((ui32Config & 0x0000ff00) == TIMER_CFG_B_ONE_SHOT) ||
00368              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_ONE_SHOT_UP) ||

00371              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_CAP_COUNT) ||
00372              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_CAP_COUNT_UP) ||
00373              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_CAP_TIME) ||
00374              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_CAP_TIME_UP) ||
00375              ((ui32Config & 0x0000ff00) == TIMER_CFG_B_PWM))));
00376 
00377     //
00378     // Disable the timers.
00379     //
00380     HWREG(ui32Base + TIMER_O_CTL) &= ~(TIMER_CTL_TAEN | TIMER_CTL_TBEN);
00381 
00382     //
00383     // Set the global timer configuration.
00384     //
00385     HWREG(ui32Base + TIMER_O_CFG) = ui32Config >> 24;
00386 
00387     //
00388     // Set the configuration of the A and B timers and set the TxPWMIE bit.
00389     // Note that the B timer configuration is ignored by the hardware in 32-bit
00390     // modes.
00391     //
00392     if(NEW_TIMER_CONFIGURATION)
00393     {
00394         HWREG(ui32Base + TIMER_O_TAMR) = (((ui32Config & 0x000f0000) >> 4) |
00395                                           (ui32Config & 0xff) |
00396                                           TIMER_TAMR_TAPWMIE);
00397         HWREG(ui32Base + TIMER_O_TBMR) = (((ui32Config & 0x00f00000) >> 8) |
00398                                           ((ui32Config >> 8) & 0xff) |
00399                                           TIMER_TBMR_TBPWMIE);
00400     }
00401     else
00402     {
00403         HWREG(ui32Base + TIMER_O_TAMR) = ((ui32Config & 0xff) |
00404                                           TIMER_TAMR_TAPWMIE);
00405         HWREG(ui32Base + TIMER_O_TBMR) = (((ui32Config >> 8) & 0xff) |
00406                                           TIMER_TBMR_TBPWMIE);
00407     }
00408 }

00471     ASSERT((ui32Timer == TIMER_A) || (ui32Timer == TIMER_B) ||
00472            (ui32Timer == TIMER_BOTH));
00473 
00474     //
00475     // On newer devices the Timer time out ADC trigger enable must also
00476     // be set.
00477     //
00478     if(NEW_TIMER_CONFIGURATION)
00479     {
00480         uint32_t ui32Val;
00481 
00482         //
00483         // Determine which bits to set or clear in GPTMADCEV.
00484         //
00485         ui32Val = (TIMER_ADCEV_TATOADCEN | TIMER_ADCEV_TBTOADCEN);
00486         ui32Val &= ui32Timer;
00487 
00488         //
00489         // Write the GPTM ADC Event register to enable or disable the trigger
00490         // to the ADC.
00491         //
00492         HWREG(ui32Base + TIMER_O_ADCEV) = (bEnable ?
00493                                            (HWREG(ui32Base + TIMER_O_ADCEV) |
00494                                             ui32Val) :
00495                                            (HWREG(ui32Base + TIMER_O_ADCEV) &
00496                                             ~(ui32Val)));
00497     }
00498 
00499     //
00500     // Set the trigger output as requested.
00501     // Set the ADC trigger output as requested.
00502     //
00503     ui32Timer &= TIMER_CTL_TAOTE | TIMER_CTL_TBOTE;
00504     HWREG(ui32Base + TIMER_O_CTL) = (bEnable ?
00505                                      (HWREG(ui32Base + TIMER_O_CTL) |
00506                                       ui32Timer) :
