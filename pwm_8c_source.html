00199 void
00200 PWMGenConfigure(uint32_t ui32Base, uint32_t ui32Gen,
00201                 uint32_t ui32Config)
00202 {
00203     //
00204     // Check the arguments.
00205     //
00206     ASSERT((ui32Base == PWM0_BASE) || (ui32Base == PWM1_BASE));
00207     ASSERT(_PWMGenValid(ui32Gen));
00208 
00209     //
00210     // Compute the generator's base address.
00211     //
00212     ui32Gen = PWM_GEN_BADDR(ui32Base, ui32Gen);
00213 
00214     //
00215     // Change the global configuration of the generator.
00216     //
00217     HWREG(ui32Gen + PWM_O_X_CTL) = ((HWREG(ui32Gen + PWM_O_X_CTL) &
00218                                      ~(PWM_X_CTL_MODE | PWM_X_CTL_DEBUG |
00219                                        PWM_X_CTL_LATCH | PWM_X_CTL_MINFLTPER |
00220                                        PWM_X_CTL_FLTSRC |
00221                                        PWM_X_CTL_DBFALLUPD_M |
00222                                        PWM_X_CTL_DBRISEUPD_M |
00223                                        PWM_X_CTL_DBCTLUPD_M |
00224                                        PWM_X_CTL_GENBUPD_M |
00225                                        PWM_X_CTL_GENAUPD_M |
00226                                        PWM_X_CTL_LOADUPD | PWM_X_CTL_CMPAUPD |
00227                                        PWM_X_CTL_CMPBUPD)) | ui32Config);
00228 
00229     //
00230     // Set the individual PWM generator controls.
00231     //
00232     if(ui32Config & PWM_X_CTL_MODE)
00233     {
00234         //
00235         // In up/down count mode, set the signal high on up count comparison
00236         // and low on down count comparison (that is, center align the
00237         // signals).
00238         //
00239         HWREG(ui32Gen + PWM_O_X_GENA) = (PWM_X_GENA_ACTCMPAU_ONE |
00240                                          PWM_X_GENA_ACTCMPAD_ZERO);
00241         HWREG(ui32Gen + PWM_O_X_GENB) = (PWM_X_GENB_ACTCMPBU_ONE |
00242                                          PWM_X_GENB_ACTCMPBD_ZERO);
00243     }
00244     else
00245     {
00246         //
00247         // In down count mode, set the signal high on load and low on count
00248         // comparison (that is, left align the signals).
00249         //
00250         HWREG(ui32Gen + PWM_O_X_GENA) = (PWM_X_GENA_ACTLOAD_ONE |
00251                                          PWM_X_GENA_ACTCMPAD_ZERO);
00252         HWREG(ui32Gen + PWM_O_X_GENB) = (PWM_X_GENB_ACTLOAD_ONE |
00253                                          PWM_X_GENB_ACTCMPBD_ZERO);
00254     }
00255 }

00279                 uint32_t ui32Period)
00280 {
00281     //
00282     // Check the arguments.
00283     //
00284     ASSERT((ui32Base == PWM0_BASE) || (ui32Base == PWM1_BASE));
00285     ASSERT(_PWMGenValid(ui32Gen));
00286 
00287     //
00288     // Compute the generator's base address.
00289     //
00290     ui32Gen = PWM_GEN_BADDR(ui32Base, ui32Gen);
00291 
00292     //
00293     // Set the reload register based on the mode.
00294     //
00295     if(HWREG(ui32Gen + PWM_O_X_CTL) & PWM_X_CTL_MODE)
00296     {
00297         //
00298         // In up/down count mode, set the reload register to half the requested
00299         // period.
00300         //
00301         ASSERT((ui32Period / 2) < 65536);
00302         HWREG(ui32Gen + PWM_O_X_LOAD) = ui32Period / 2;
00303     }
00304     else
00305     {
00306         //
00307         // In down count mode, set the reload register to the requested period
00308         // minus one.
00309         //
00310         ASSERT((ui32Period <= 65536) && (ui32Period != 0));
00311         HWREG(ui32Gen + PWM_O_X_LOAD) = ui32Period - 1;
00312     }
00313 }

00446 void
00447 PWMPulseWidthSet(uint32_t ui32Base, uint32_t ui32PWMOut,
00448                  uint32_t ui32Width)
00449 {
00450     uint32_t ui32GenBase, ui32Reg;
00451 
00452     //
00453     // Check the arguments.
00454     //
00455     ASSERT((ui32Base == PWM0_BASE) || (ui32Base == PWM1_BASE));
00456     ASSERT(_PWMOutValid(ui32PWMOut));
00457 
00458     //
00459     // Compute the generator's base address.
00460     //
00461     ui32GenBase = PWM_OUT_BADDR(ui32Base, ui32PWMOut);
00462 
00463     //
00464     // If the counter is in up/down count mode, divide the width by two.
00465     //
00466     if(HWREG(ui32GenBase + PWM_O_X_CTL) & PWM_X_CTL_MODE)
00467     {
00468         ui32Width /= 2;
00469     }
00470 
00471     //
00472     // Get the period.
00473     //
00474     ui32Reg = HWREG(ui32GenBase + PWM_O_X_LOAD);
00475 
00476     //
00477     // Make sure the width is not too large.
00478     //
00479     ASSERT(ui32Width < ui32Reg);
00480 
00481     //
00482     // Compute the compare value.
00483     //
00484     ui32Reg = ui32Reg - ui32Width;
00485 
00486     //
00487     // Write to the appropriate registers.
00488     //
00489     if(PWM_IS_OUTPUT_ODD(ui32PWMOut))
00490     {
00491         HWREG(ui32GenBase + PWM_O_X_CMPB) = ui32Reg;
00492     }
00493     else
00494     {
00495         HWREG(ui32GenBase + PWM_O_X_CMPA) = ui32Reg;
00496     }
00497 }

00517 uint32_t
00518 PWMPulseWidthGet(uint32_t ui32Base, uint32_t ui32PWMOut)
00519 {
00520     uint32_t ui32GenBase, ui32Reg, ui32Load;
00521 
00522     //
00523     // Check the arguments.
00524     //
00525     ASSERT((ui32Base == PWM0_BASE) || (ui32Base == PWM1_BASE));
00526     ASSERT(_PWMOutValid(ui32PWMOut));
00527 
00528     //
00529     // Compute the generator's base address.
00530     //
00531     ui32GenBase = PWM_OUT_BADDR(ui32Base, ui32PWMOut);
00532 
00533     //
00534     // Then compute the pulse width.  If mode is UpDown, set
00535     // width = (load - compare) * 2.  Otherwise, set width = load - compare.
00536     //
00537     ui32Load = HWREG(ui32GenBase + PWM_O_X_LOAD);
00538     if(PWM_IS_OUTPUT_ODD(ui32PWMOut))
00539     {
00540         ui32Reg = HWREG(ui32GenBase + PWM_O_X_CMPB);
00541     }
00542     else
00543     {
00544         ui32Reg = HWREG(ui32GenBase + PWM_O_X_CMPA);
00545     }
00546     ui32Reg = ui32Load - ui32Reg;
00547 
00548     //
00549     // If in up/down count mode, double the pulse width.
00550     //
00551     if(HWREG(ui32GenBase + PWM_O_X_CTL) & PWM_X_CTL_MODE)
00552     {
00553         ui32Reg = ui32Reg * 2;
00554     }
00555 
00556     //
00557     // Return the pulse width.
00558     //
00559     return(ui32Reg);
00560 }
