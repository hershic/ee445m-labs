00057 /*
00058   This method handles the drawing of each part of a graph object. The
00059   task of drawing each part has been delegated to a subroutine for
00060   ease of debugging and testing. Before calling each subroutine, this
00061   method sets the internal framebuffer variables inside of the
00062   specified graph object.
00063   Inputs:  fb                framebuffer to use as canvas
00064   g                 graph to draw on fb
00065   top_left_corner   top left corner of graph

00068   Outputs: void
00069 
00070   ** NOTICE **
00071   This is where the rubber meets the road.
00072 */
00073 void FBDrawGraph(framebuffer fb, graph* g, point* top_left_corner,
00074                  pixel_t width, pixel_t height) {
00075   /* Determine how many pixels to reserve for the scale numbers next
00076    * to the axes */
00077   pixel_t x_range_padding = max_pixel_width_of_long(g->y_min, g->y_max);
00078   pixel_t y_range_padding = max_pixel_height_of_long(g->x_min, g->x_max);
00079 
00080   /* {x,y}_pixel_{start,end} information in /resources/2014-04-07 10.07.01.jpg */
00081   g->x_pixel_start = top_left_corner->x + x_range_padding;
00082   g->x_pixel_end   = top_left_corner->x + width;
00083   g->y_pixel_start = top_left_corner->y;
00084   g->y_pixel_end   = top_left_corner->y + height - 2 - y_range_padding;
00085 
00086   _FBDrawGraphBounds(fb, g, x_range_padding, y_range_padding, top_left_corner, width, height, top_left_corner->shade);
00087   _FBDrawGraphData(fb, g, top_left_corner, top_left_corner->shade);
00088   _FBDrawGraphTitle(fb, g);
