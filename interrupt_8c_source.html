00310 {
00311     uint32_t ui32Idx, ui32Value;
00312 
00313     //
00314     // Check the arguments.
00315     //
00316     ASSERT(ui32Interrupt < NUM_INTERRUPTS);
00317 
00318     //
00319     // Make sure that the RAM vector table is correctly aligned.
00320     //
00321     ASSERT(((uint32_t)g_pfnRAMVectors & 0x000003ff) == 0);
00322 
00323     //
00324     // See if the RAM vector table has been initialized.
00325     //
00326     if(HWREG(NVIC_VTABLE) != (uint32_t)g_pfnRAMVectors)
00327     {
00328         //
00329         // Copy the vector table from the beginning of FLASH to the RAM vector
00330         // table.
00331         //
00332         ui32Value = HWREG(NVIC_VTABLE);
00333         for(ui32Idx = 0; ui32Idx < NUM_INTERRUPTS; ui32Idx++)
00334         {
00335             g_pfnRAMVectors[ui32Idx] = (void (*)(void))HWREG((ui32Idx * 4) +
00336                                                              ui32Value);
00337         }
00338 
00339         //
00340         // Point the NVIC at the RAM vector table.
00341         //
00342         HWREG(NVIC_VTABLE) = (uint32_t)g_pfnRAMVectors;
00343     }
00344 
00345     //
00346     // Save the interrupt handler.
00347     //
00348     g_pfnRAMVectors[ui32Interrupt] = pfnHandler;

00610 IntEnable(uint32_t ui32Interrupt)
00611 {
00612     //
00613     // Check the arguments.
00614     //
00615     ASSERT(ui32Interrupt < NUM_INTERRUPTS);
00616 
00617     //
00618     // Determine the interrupt to enable.
00619     //
00620     if(ui32Interrupt == FAULT_MPU)
00621     {
00622         //
00623         // Enable the MemManage interrupt.
00624         //
00625         HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_MEM;
00626     }
00627     else if(ui32Interrupt == FAULT_BUS)
00628     {
00629         //
00630         // Enable the bus fault interrupt.
00631         //
00632         HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_BUS;
00633     }
00634     else if(ui32Interrupt == FAULT_USAGE)
00635     {
00636         //
00637         // Enable the usage fault interrupt.
00638         //
00639         HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_USAGE;
00640     }
00641     else if(ui32Interrupt == FAULT_SYSTICK)
00642     {
00643         //
00644         // Enable the System Tick interrupt.
00645         //
00646         HWREG(NVIC_ST_CTRL) |= NVIC_ST_CTRL_INTEN;
00647     }
00648     else if(ui32Interrupt >= 16)
00649     {
00650         //
00651         // Enable the general interrupt.
00652         //
00653         HWREG(g_pui32EnRegs[(ui32Interrupt - 16) / 32]) =
00654             1 << ((ui32Interrupt - 16) & 31);
00655     }
00656 }
