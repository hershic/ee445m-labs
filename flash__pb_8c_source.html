00098 FlashPBIsValid(uint8_t *pui8Offset)
00099 {
00100     uint32_t ui32Idx, ui32Sum;
00101 
00102     //
00103     // Check the arguments.
00104     //
00105     ASSERT(pui8Offset != (void *)0);
00106 
00107     //
00108     // Loop through the bytes in the block, computing the checksum.
00109     //
00110     for(ui32Idx = 0, ui32Sum = 0; ui32Idx < g_ui32FlashPBSize; ui32Idx++)
00111     {
00112         ui32Sum += pui8Offset[ui32Idx];
00113     }
00114 
00115     //
00116     // The checksum should be zero, so return a failure if it is not.
00117     //
00118     if((ui32Sum & 255) != 0)
00119     {
00120         return(0);
00121     }
00122 
00123     //
00124     // If the sum is equal to the size * 255, then the block is all ones and
00125     // should not be considered valid.
00126     //
00127     if((g_ui32FlashPBSize * 255) == ui32Sum)
00128     {
00129         return(0);
00130     }
00131 
00132     //
00133     // This is a valid parameter block.
00134     //
00135     return(1);
00136 }

00199 void
00200 FlashPBSave(uint8_t *pui8Buffer)
00201 {
00202     uint8_t *pui8New;
00203     uint32_t ui32Idx, ui32Sum;
00204 
00205     //
00206     // Check the arguments.
00207     //
00208     ASSERT(pui8Buffer != (void *)0);
00209 
00210     //
00211     // See if there is a valid parameter block in flash.
00212     //
00213     if(g_pui8FlashPBCurrent)
00214     {
00215         //
00216         // Set the sequence number to one greater than the most recent
00217         // parameter block.
00218         //
00219         pui8Buffer[0] = g_pui8FlashPBCurrent[0] + 1;
00220 
00221         //
00222         // Try to write the new parameter block immediately after the most
00223         // recent parameter block.
00224         //
00225         pui8New = g_pui8FlashPBCurrent + g_ui32FlashPBSize;
00226         if(pui8New == g_pui8FlashPBEnd)
00227         {
00228             pui8New = g_pui8FlashPBStart;
00229         }
00230     }
00231     else
00232     {
00233         //
00234         // There is not a valid parameter block in flash, so set the sequence
00235         // number of this parameter block to zero.
00236         //
00237         pui8Buffer[0] = 0;
00238 
00239         //
00240         // Try to write the new parameter block at the beginning of the flash
00241         // space for parameter blocks.
00242         //
00243         pui8New = g_pui8FlashPBStart;
00244     }
00245 
00246     //
00247     // Compute the checksum of the parameter block to be written.
00248     //
00249     for(ui32Idx = 0, ui32Sum = 0; ui32Idx < g_ui32FlashPBSize; ui32Idx++)
00250     {
00251         ui32Sum -= pui8Buffer[ui32Idx];
00252     }
00253 
00254     //
00255     // Store the checksum into the parameter block.
00256     //
00257     pui8Buffer[1] += ui32Sum;
00258 
00259     //
00260     // Look for a location to store this parameter block.  This infinite loop
00261     // will be explicitly broken out of when a valid location is found.
00262     //
00263     while(1)
00264     {
00265         //
00266         // See if this location is at the start of an erase block.
00267         //
00268         if(((uint32_t)pui8New & (FLASH_SECTOR_SIZE - 1)) == 0)
00269         {
00270             //
00271             // Erase this block of the flash.  This does not assume that the
00272             // erase succeeded in case this block of the flash has become bad
00273             // through too much use.  Given the extremely low frequency that
00274             // the parameter blocks are written, this will likely never fail.
00275             // But, that assumption is not made in order to be safe.
00276             //
00277             MAP_FlashErase((uint32_t)pui8New);
00278         }
00279 
00280         //
00281         // Loop through this portion of flash to see if is all ones (in other
00282         // words, it is an erased portion of flash).
00283         //
00284         for(ui32Idx = 0; ui32Idx < g_ui32FlashPBSize; ui32Idx++)
00285         {
00286             if(pui8New[ui32Idx] != 0xff)
00287             {
00288                 break;
00289             }
00290         }
00291 
00292         //
00293         // If all bytes in this portion of flash are ones, then break out of
00294         // the loop since this is a good location for storing the parameter
00295         // block.
00296         //
00297         if(ui32Idx == g_ui32FlashPBSize)
00298         {
00299             break;
00300         }
00301 
00302         //
00303         // Increment to the next parameter block location.
00304         //
00305         pui8New += g_ui32FlashPBSize;
00306         if(pui8New == g_pui8FlashPBEnd)
00307         {
00308             pui8New = g_pui8FlashPBStart;
00309         }
00310 
00311         //
00312         // If every possible location has been checked and none are valid, then
00313         // it will not be possible to write this parameter block.  Simply
00314         // return without writing it.
00315         //
00316         if((g_pui8FlashPBCurrent && (pui8New == g_pui8FlashPBCurrent)) ||
00317            (!g_pui8FlashPBCurrent && (pui8New == g_pui8FlashPBStart)))
00318         {
00319             return;
00320         }
00321     }
00322 
00323     //
00324     // Write this parameter block to flash.
00325     //
00326     MAP_FlashProgram((uint32_t *)pui8Buffer, (uint32_t)pui8New,
00327                      g_ui32FlashPBSize);
00328 
00329     //
00330     // Compare the parameter block data to the data that should now be in
00331     // flash.  Return if any of the data does not compare, leaving the previous
00332     // parameter block in flash as the most recent (since the current parameter
00333     // block failed to properly program).
00334     //
00335     for(ui32Idx = 0; ui32Idx < g_ui32FlashPBSize; ui32Idx++)
00336     {
00337         if(pui8New[ui32Idx] != pui8Buffer[ui32Idx])
00338         {
00339             return;
00340         }
00341     }
00342 
00343     //
00344     // The new parameter block becomes the most recent parameter block.
00345     //
00346     g_pui8FlashPBCurrent = pui8New;
00347 }

00410 FlashPBInit(uint32_t ui32Start, uint32_t ui32End, uint32_t ui32Size)
00411 {

00413     uint8_t ui8One, ui8Two;
00414 
00415     //
00416     // Check the arguments.
00417     //
00418     ASSERT((ui32Start % FLASH_SECTOR_SIZE) == 0);
00419     ASSERT((ui32End % FLASH_SECTOR_SIZE) == 0);
00420     ASSERT((FLASH_SECTOR_SIZE % ui32Size) == 0);
00421 
00422     //
00423     // Save the characteristics of the flash memory to be used for storing
00424     // parameter blocks.
00425     //
00426     g_pui8FlashPBStart = (uint8_t *)ui32Start;
00427     g_pui8FlashPBEnd = (uint8_t *)ui32End;
00428     g_ui32FlashPBSize = ui32Size;
00429 
00430     //
00431     // Loop through the portion of flash memory used for storing parameter
00432     // blocks.
00433     //
00434     for(pui8Offset = g_pui8FlashPBStart, pui8Current = 0;
00435         pui8Offset < g_pui8FlashPBEnd; pui8Offset += g_ui32FlashPBSize)
00436     {
00437         //
00438         // See if this is a valid parameter block (in other words, the checksum
00439         // is correct).
00440         //
00441         if(FlashPBIsValid(pui8Offset))
00442         {
00443             //
00444             // See if a valid parameter block has been previously found.
00445             //
00446             if(pui8Current != 0)
00447             {
00448                 //
00449                 // Get the sequence numbers for the current and new parameter
00450                 // blocks.
00451                 //
00452                 ui8One = pui8Current[0];
00453                 ui8Two = pui8Offset[0];
00454 
00455                 //
00456                 // See if the sequence number for the new parameter block is
00457                 // greater than the current block.  The comparison isn't
00458                 // straightforward since the one byte sequence number will wrap
00459                 // after 256 parameter blocks.
00460                 //
00461                 if(((ui8One > ui8Two) && ((ui8One - ui8Two) < 128)) ||
00462                    ((ui8Two > ui8One) && ((ui8Two - ui8One) > 128)))
00463                 {
00464                     //
00465                     // The new parameter block is older than the current
00466                     // parameter block, so skip the new parameter block and
00467                     // keep searching.
00468                     //
00469                     continue;
00470                 }
00471             }
00472 
00473             //
00474             // The new parameter block is more recent than the current one, so
00475             // make it the new current parameter block.
00476             //
00477             pui8Current = pui8Offset;
00478         }
00479     }
00480 
00481     //
00482     // Save the address of the most recent parameter block found.  If no valid
00483     // parameter blocks were found, this will be a NULL pointer.
00484     //
00485     g_pui8FlashPBCurrent = pui8Current;
