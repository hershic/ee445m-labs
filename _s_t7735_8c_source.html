460 // The Data/Command pin must be valid when the eighth bit is
461 // sent. The SSI module has hardware input and output FIFOs
462 // that are 8 locations deep. Based on the observation that
463 // the LCD interface tends to send a few commands and then a
464 // lot of data, the FIFOs are not used when writing
465 // commands, and they are used when writing data. This
466 // ensures that the Data/Command pin status matches the byte
467 // that is actually being transmitted.
468 // The write command operation waits until all data has been
469 // sent, configures the Data/Command pin for commands, sends
470 // the command, and then waits for the transmission to
471 // finish.
472 // The write data operation waits until there is room in the
473 // transmit FIFO, configures the Data/Command pin for data,
474 // and then adds the data to the transmit FIFO.
475 // NOTE: These functions will crash or stall indefinitely if
476 // the SSI0 module is not initialized and enabled.
477 void static writecommand(uint8_t c) {
478  // wait until SSI0 not busy/transmit FIFO empty
479  while((SSI0_SR_R&SSI_SR_BSY)==SSI_SR_BSY){};
480  DC = DC_COMMAND;
481  SSI0_DR_R = c; // data out
482  // wait until SSI0 not busy/transmit FIFO empty
483  while((SSI0_SR_R&SSI_SR_BSY)==SSI_SR_BSY){};
484 }

488  while((SSI0_SR_R&SSI_SR_TNF)==0){}; // wait until transmit FIFO not full
489  DC = DC_DATA;

491 }

645 // Companion code to the above tables. Reads and issues
646 // a series of LCD commands stored in ROM byte array.
647 void static commandList(const uint8_t *addr) {
648 
649  uint8_t numCommands, numArgs;
650  uint16_t ms;
651 
652  numCommands = *(addr++); // Number of commands to follow
653  while(numCommands--) { // For each command...
654  writecommand(*(addr++)); // Read, issue command
655  numArgs = *(addr++); // Number of args to follow
656  ms = numArgs & DELAY; // If hibit set, delay follows args
657  numArgs &= ~DELAY; // Mask out delay bit
658  while(numArgs--) { // For each argument...
659  writedata(*(addr++)); // Read, issue argument
660  }
661 
662  if(ms) {
663  ms = *(addr++); // Read post-command delay time (ms)
664  if(ms == 255) ms = 500; // If 255, delay for 500 ms
665  Delay1ms(ms);
666  }
667  }
668 }

671 // Initialization code common to both 'B' and 'R' type displays
672 void static commonInit(const uint8_t *cmdList) {
673  volatile uint32_t delay;
674  ColStart = RowStart = 0; // May be overridden in init func
675 
676  SYSCTL_RCGCSSI_R |= 0x01; // activate SSI0
677  SYSCTL_RCGCGPIO_R |= 0x01; // activate port A
678  while((SYSCTL_PRGPIO_R&0x01)==0){}; // allow time for clock to start
679 
680  // toggle RST low to reset; CS low so it'll listen to us
681  // SSI0Fss is temporarily used as GPIO
682  GPIO_PORTA_DIR_R |= 0xC8; // make PA3,6,7 out
683  GPIO_PORTA_AFSEL_R &= ~0xC8; // disable alt funct on PA3,6,7
684  GPIO_PORTA_DEN_R |= 0xC8; // enable digital I/O on PA3,6,7
685  // configure PA3,6,7 as GPIO
686  GPIO_PORTA_PCTL_R = (GPIO_PORTA_PCTL_R&0x00FF0FFF)+0x00000000;
687  GPIO_PORTA_AMSEL_R &= ~0xC8; // disable analog functionality on PA3,6,7
688  TFT_CS = TFT_CS_LOW;
689  RESET = RESET_HIGH;
690  Delay1ms(500);
691  RESET = RESET_LOW;
692  Delay1ms(500);
693  RESET = RESET_HIGH;
694  Delay1ms(500);
695 
696  // initialize SSI0
697  GPIO_PORTA_AFSEL_R |= 0x2C; // enable alt funct on PA2,3,5
698  GPIO_PORTA_DEN_R |= 0x2C; // enable digital I/O on PA2,3,5
699  // configure PA2,3,5 as SSI
700  GPIO_PORTA_PCTL_R = (GPIO_PORTA_PCTL_R&0xFF0F00FF)+0x00202200;
701  GPIO_PORTA_AMSEL_R &= ~0x2C; // disable analog functionality on PA2,3,5
702  SSI0_CR1_R &= ~SSI_CR1_SSE; // disable SSI
703  SSI0_CR1_R &= ~SSI_CR1_MS; // master mode
704  // configure for system clock/PLL baud clock source
705  SSI0_CC_R = (SSI0_CC_R&~SSI_CC_CS_M)+SSI_CC_CS_SYSPLL;
706 // // clock divider for 3.125 MHz SSIClk (50 MHz PIOSC/16)
707 // SSI0_CPSR_R = (SSI0_CPSR_R&~SSI_CPSR_CPSDVSR_M)+16;
708  // clock divider for 8 MHz SSIClk (80 MHz PLL/24)
709  // SysClk/(CPSDVSR*(1+SCR))
710  // 80/(10*(1+0)) = 8 MHz (slower than 4 MHz)
711  SSI0_CPSR_R = (SSI0_CPSR_R&~SSI_CPSR_CPSDVSR_M)+10; // must be even number
712  SSI0_CR0_R &= ~(SSI_CR0_SCR_M | // SCR = 0 (8 Mbps data rate)
713  SSI_CR0_SPH | // SPH = 0
714  SSI_CR0_SPO); // SPO = 0
715  // FRF = Freescale format
716  SSI0_CR0_R = (SSI0_CR0_R&~SSI_CR0_FRF_M)+SSI_CR0_FRF_MOTO;
717  // DSS = 8-bit data
718  SSI0_CR0_R = (SSI0_CR0_R&~SSI_CR0_DSS_M)+SSI_CR0_DSS_8;
719  SSI0_CR1_R |= SSI_CR1_SSE; // enable SSI
720 
721  if(cmdList) commandList(cmdList);
722 }

926 // with dummy columns so the data have four byte alignment. This
927 // function assumes that all of that has been stripped out, and the

932 // converter program.

934 // Requires (11 + 2*w*h) bytes of transmission (assuming image fully on screen)
935 // Input: x horizontal position of the bottom left corner of the image, columns from the left edge
936 // y vertical position of the bottom left corner of the image, rows from the top edge
937 // image pointer to a 16-bit color BMP image
938 // w number of pixels wide
939 // h number of pixels tall

942 void ST7735_DrawBitmap(int16_t x, int16_t y, const uint16_t *image, int16_t w, int16_t h){

944  int16_t originalWidth = w; // save this value; even if not all columns fit on the screen, the image is still this width in ROM

946

949  }

952  //following logic much more complicated, since you can have
953  //an image that exceeds multiple boundaries and needs to be
954  //clipped on more than one side.
955  return;

958  skipC = (x + w) - _width; // skip cut off columns
959  w = _width - x;
960  }
961  if((y - h + 1) < 0){ // image exceeds top of screen
962  i = i - (h - y - 1)*originalWidth; // skip the last cut off rows
963  h = y + 1;

965  if(x < 0){ // image exceeds left of screen
966  w = w + x;
967  skipC = -1*x; // skip cut off columns
968  i = i - x; // skip the first cut off columns
969  x = 0;
970  }
971  if(y >= _height){ // image exceeds bottom of screen
972  h = h - (y - _height + 1);
973  y = _height - 1;
974  }

981  writedata((uint8_t)(image[i] >> 8));
982  // send the bottom 8 bits
983  writedata((uint8_t)image[i]);
984  i = i + 1; // go to the next pixel
985  }

1043 // from Adafruit_GFX.c but adapted for this processor. However, this
1044 // function only uses one call to setAddrWindow(), which allows it to
1045 // run at least twice as fast.
1046 // Requires (11 + size*size*6*8) bytes of transmission (assuming image fully on screen)
1047 // Input: x horizontal position of the top left corner of the character, columns from the left edge
1048 // y vertical position of the top left corner of the character, rows from the top edge
1049 // c character to be printed
1050 // textColor 16-bit color of the character
1051 // bgColor 16-bit color of the background
1052 // size number of pixels per character pixel (e.g. size==2 prints each pixel of font as 2x2 square)
1053 // Output: none
1054 void ST7735_DrawChar(int16_t x, int16_t y, uint8_t c, int16_t textColor, int16_t bgColor, uint8_t size){

1056  int32_t col, row, i, j;// loop indices
1057  if(((x + 5*size - 1) >= _width) || // Clip right
1058  ((y + 8*size - 1) >= _height) || // Clip bottom
1059  ((x + 5*size - 1) < 0) || // Clip left
1060  ((y + 8*size - 1) < 0)){ // Clip top
1061  return;
1062  }
1063

1073  // bit is set in Font, print pixel(s) in text color

1087  }
1088  }
