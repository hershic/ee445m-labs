00266 Crc8CCITT(uint8_t ui8Crc, const uint8_t *pui8Data, uint32_t ui32Count)
00267 {
00268     uint32_t ui32Temp;
00269 
00270     //
00271     // If the data buffer is not 16 bit-aligned, then perform a single step of
00272     // the CRC to make it 16 bit-aligned.
00273     //
00274     if((uint32_t)pui8Data & 1)
00275     {
00276         //
00277         // Perform the CRC on this input byte.
00278         //
00279         ui8Crc = CRC8_ITER(ui8Crc, *pui8Data);
00280 
00281         //
00282         // Skip this input byte.
00283         //
00284         pui8Data++;
00285         ui32Count--;
00286     }
00287 
00288     //
00289     // If the data buffer is not word-aligned and there are at least two bytes
00290     // of data left, then perform two steps of the CRC to make it word-aligned.
00291     //
00292     if(((uint32_t)pui8Data & 2) && (ui32Count > 1))
00293     {
00294         //
00295         // Read the next 16 bits.
00296         //
00297         ui32Temp = *(uint16_t *)pui8Data;
00298 
00299         //
00300         // Perform the CRC on these two bytes.
00301         //
00302         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp);
00303         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp >> 8);
00304 
00305         //
00306         // Skip these input bytes.
00307         //
00308         pui8Data += 2;
00309         ui32Count -= 2;
00310     }
00311 
00312     //
00313     // While there is at least a word remaining in the data buffer, perform
00314     // four steps of the CRC to consume a word.
00315     //
00316     while(ui32Count > 3)
00317     {
00318         //
00319         // Read the next word.
00320         //
00321         ui32Temp = *(uint32_t *)pui8Data;
00322 
00323         //
00324         // Perform the CRC on these four bytes.
00325         //
00326         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp);
00327         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp >> 8);
00328         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp >> 16);
00329         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp >> 24);
00330 
00331         //
00332         // Skip these input bytes.
00333         //
00334         pui8Data += 4;
00335         ui32Count -= 4;
00336     }
00337 
00338     //
00339     // If there are 16 bits left in the input buffer, then perform two steps of
00340     // the CRC.
00341     //
00342     if(ui32Count > 1)
00343     {
00344         //
00345         // Read the 16 bits.
00346         //
00347         ui32Temp = *(uint16_t *)pui8Data;
00348 
00349         //
00350         // Perform the CRC on these two bytes.
00351         //
00352         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp);
00353         ui8Crc = CRC8_ITER(ui8Crc, ui32Temp >> 8);
00354 
00355         //
00356         // Skip these input bytes.
00357         //
00358         pui8Data += 2;
00359         ui32Count -= 2;
00360     }
00361

00363     // If there is a final byte remaining in the input buffer, then perform a
00364     // single step of the CRC.
00365     //
00366     if(ui32Count != 0)
00367     {
00368         ui8Crc = CRC8_ITER(ui8Crc, *pui8Data);

00370

00373     //
