00109 uint32_t
00110 LCDModeSet(uint32_t ui32Base, uint8_t ui8Mode, uint32_t ui32PixClk,
00111            uint32_t ui32SysClk)
00112 {
00113     uint32_t ui32Div;
00114 
00115     //
00116     // Sanity check parameters.
00117     //
00118     ASSERT(ui32Base == LCD0_BASE);
00119     ASSERT((ui8Mode & ~(LCD_MODE_RASTER | LCD_MODE_LIDD |
00120                         LCD_MODE_AUTO_UFLOW_RESTART)) == 0);
00121 
00122     //
00123     // Enable clocks to the LCD controller submodules.
00124     //
00125     HWREG(ui32Base + LCD_O_CLKEN) = (LCD_CLKEN_DMA | LCD_CLKEN_CORE |
00126                                      LCD_CLKEN_LIDD);
00127 
00128     //
00129     // Determine the clock divider to use to get as close as possible to the
00130     // desired pixel clock.  Note that we set the division up so that we
00131     // round the divisor up and ensure that the clock used is never faster
00132     // than the requested rate.
00133     //
00134     ui32Div = (ui32SysClk + (ui32PixClk - 1)) / ui32PixClk;
00135 
00136     //
00137     // Check that the calculated value is valid.
00138     //
00139     ASSERT(ui32Div);
00140     ASSERT(ui32Div < 256);
00141     ASSERT(!((ui8Mode & LCD_MODE_RASTER) && (ui32Div < 2)));
00142 
00143     //
00144     // Write the LCDCTL register to set the mode.
00145     //
00146     HWREG(ui32Base + LCD_O_CTL) = (uint32_t)ui8Mode |
00147                                   (ui32Div << LCD_CTL_CLKDIV_S);
00148 
00149     //
00150     // Return the selected clock rate.  Finding ui32Div set to 0 should not
00151     // happen unless someone passed pathological arguments and builds without
00152     // the ASSERTS, but we guard against it just in case.
00153     //
00154     return(ui32Div ? (ui32SysClk / ui32Div) : ui32SysClk);
00155 }

00182 {

00720 void
00721 LCDIDDDMAWrite(uint32_t ui32Base, uint32_t ui32CS, const uint32_t *pui32Data,
00722                uint32_t ui32Count)
00723 {
00724     //
00725     // Sanity check parameters.
00726     //
00727     ASSERT(ui32Base == LCD0_BASE);
00728     ASSERT((ui32CS == 0) || (ui32CS == 1));
00729     ASSERT(!((uint32_t)pui32Data & 3));
00730     ASSERT(!(ui32Count & 1));
00731 
00732     //
00733     // Make sure DMA is disabled so that enabling it triggers this new
00734     // transfer.
00735     //
00736     HWREG(ui32Base + LCD_O_LIDDCTL) &= ~LCD_LIDDCTL_DMAEN;
00737 
00738     //
00739     // Set up the transfer.  Note that the ceiling register must contain the
00740     // address of the last word which contains data we want transfered and NOT
00741     // the first location after the data we want written.
00742     //
00743     HWREG(ui32Base + LCD_O_DMABAFB0) = (uint32_t)pui32Data;
00744     HWREG(ui32Base + LCD_O_DMACAFB0) = ((uint32_t)pui32Data +
00745                                         (ui32Count * 2) - 4);
00746 
00747     //
00748     // Tell the controller which CS to use for the DMA transaction.
00749     //
00750     if(!ui32CS)
00751     {
00752         //
00753         // Use CS0.
00754         //
00755         HWREG(ui32Base + LCD_O_LIDDCTL) &= ~LCD_LIDDCTL_DMACS;
00756     }
00757     else
00758     {
00759         //
00760         // Use CS1.
00761         //
00762         HWREG(ui32Base + LCD_O_LIDDCTL) |= LCD_LIDDCTL_DMACS;
00763     }
00764 
00765     //
00766     // Enable the DMA engine and start the transaction.
00767     //
00768     HWREG(ui32Base + LCD_O_LIDDCTL) |= LCD_LIDDCTL_DMAEN;
00769 }

00893 void
00894 LCDRasterTimingSet(uint32_t ui32Base, const tLCDRasterTiming *pTiming)
00895 {
00896     uint32_t ui32T0, ui32T1, ui32T2;
00897 
00898     //
00899     // Sanity check parameters.
00900     //
00901     ASSERT(ui32Base == LCD0_BASE);
00902     ASSERT(pTiming);
00903     ASSERT(!(pTiming->ui32Flags & ~(RASTER_TIMING_SYNCS_OPPOSITE_PIXCLK |
00904                                     RASTER_TIMING_SYNCS_ON_FALLING_PIXCLK |
00905                                     RASTER_TIMING_SYNCS_ON_RISING_PIXCLK |
00906                                     RASTER_TIMING_ACTIVE_LOW_OE |
00907                                     RASTER_TIMING_ACTIVE_LOW_PIXCLK |
00908                                     RASTER_TIMING_ACTIVE_LOW_HSYNC |
00909                                     RASTER_TIMING_ACTIVE_LOW_VSYNC)));
00910     ASSERT(pTiming->ui16PanelWidth && (pTiming->ui16PanelWidth <= 2048) &&
00911            ((pTiming->ui16PanelWidth % 16) == 0));
00912     ASSERT(pTiming->ui16PanelHeight && (pTiming->ui16PanelHeight <= 2048));
00913     ASSERT(pTiming->ui16HFrontPorch && (pTiming->ui16HFrontPorch <= 1024));
00914     ASSERT(pTiming->ui16HBackPorch && (pTiming->ui16HBackPorch <= 1024));
00915     ASSERT(pTiming->ui16HSyncWidth && (pTiming->ui16HSyncWidth <= 1024));
00916     ASSERT(pTiming->ui8VSyncWidth && (pTiming->ui8VSyncWidth <= 64));
00917 
00918     //
00919     // Construct the values we need for the three raster timing registers.
00920     //
00921     ui32T0 = ((uint32_t)((pTiming->ui16HBackPorch - 1) & 0xFF) <<
00922               LCD_RASTRTIM0_HBP_S) |
00923              ((uint32_t)((pTiming->ui16HFrontPorch - 1) & 0xFF) <<
00924               LCD_RASTRTIM0_HFP_S) |
00925              ((uint32_t)((pTiming->ui16HSyncWidth - 1) & 0x3F) <<
00926               LCD_RASTRTIM0_HSW_S) |
00927              (((uint32_t)((pTiming->ui16PanelWidth - 1) & 0x3F0) >> 4) <<
00928               LCD_RASTRTIM0_PPL_S) |
00929              (((uint32_t)((pTiming->ui16PanelWidth - 1) & 0x400) >> 10) <<
00930               LCD_RASTRTIM0_MSBPPL_S);
00931     ui32T1 = ((uint32_t)pTiming->ui8VBackPorch << LCD_RASTRTIM1_VBP_S) |
00932              ((uint32_t)pTiming->ui8VFrontPorch << LCD_RASTRTIM1_VFP_S) |
00933              ((uint32_t)((pTiming->ui8VSyncWidth - 1) & 0x3F) <<
00934               LCD_RASTRTIM1_VSW_S) |
00935              ((uint32_t)(pTiming->ui16PanelHeight - 1) & 0x3FF) <<
00936              LCD_RASTRTIM1_LPP_S;
00937     ui32T2 = pTiming->ui32Flags |
00938              ((((pTiming->ui16HSyncWidth - 1) & 0x3C0) >> 6) <<
00939               LCD_RASTRTIM2_HSW_S) |
00940              ((((pTiming->ui16PanelHeight - 1) & 0x400) >> 10) <<
00941               LCD_RASTRTIM2_MSBLPP_S) |
00942              ((((pTiming->ui16HBackPorch - 1) & 0x300) >> 8) <<
00943               LCD_RASTRTIM2_MSBHBP_S) |
00944              ((((pTiming->ui16HFrontPorch - 1) & 0x300) >> 8) <<
00945               LCD_RASTRTIM2_MSBHFP_S) |
00946              (pTiming->ui8ACBiasLineCount << LCD_RASTRTIM2_ACBF_S);
00947 
00948     //
00949     // Write the timing registers, taking care to preserve any existing value
00950     // in the AC Bias interrupt field of RASTRTIM2.
00951     //
00952     HWREG(ui32Base + LCD_O_RASTRTIM0) = ui32T0;
00953     HWREG(ui32Base + LCD_O_RASTRTIM1) = ui32T1;
00954     HWREG(ui32Base + LCD_O_RASTRTIM2) = (HWREG(ui32Base + LCD_O_RASTRTIM2) &
00955                                          LCD_RASTRTIM2_ACBI_M) | ui32T2;
00956 }

01366 void
01367 LCDRasterPaletteSet(uint32_t ui32Base, uint32_t ui32Type, uint32_t *pui32Addr,
01368                     const uint32_t *pui32SrcColors, uint32_t ui32Start,
01369                     uint32_t ui32Count)
01370 {
01371     uint16_t *pui16Pal;
01372     uint16_t *pui16Src;
01373     uint32_t ui32Loop;
01374 
01375     //
01376     // Sanity check parameters.
01377     //
01378     ASSERT(ui32Base == LCD0_BASE);
01379     ASSERT(ui32Start < 256);
01380     ASSERT((ui32Start + ui32Count) <= 256);
01381     ASSERT(pui32Addr);
01382     ASSERT((pui32SrcColors) || (ui32Count == 0));
01383     ASSERT(!(ui32Type & ~(LCD_PALETTE_SRC_24BIT | LCD_PALETTE_TYPE_DIRECT |
01384                           LCD_PALETTE_TYPE_8BPP | LCD_PALETTE_TYPE_4BPP |
01385                           LCD_PALETTE_TYPE_2BPP | LCD_PALETTE_TYPE_1BPP)));
01386 
01387     //
01388     // Get a pointer to the start of the palette.
01389     //
01390     pui16Pal = (uint16_t *)pui32Addr;
01391 
01392     //
01393     // Are we converting the palette color format?
01394     //
01395     if(ui32Type & LCD_PALETTE_SRC_24BIT)
01396     {
01397         //
01398         // Yes - loop through each of the supplied 24-bit colors converting
01399         // and storing each.
01400         //
01401         ui32Loop = 0;
01402         while(ui32Count)
01403         {
01404             pui16Pal[ui32Start + ui32Loop] =
01405                 PAL_FROM_RGB(pui32SrcColors[ui32Loop]);
01406             ui32Loop++;
01407             ui32Count--;
01408         }
01409     }
01410     else
01411     {
01412         //
01413         // No - loop through the supplied 12-bit colors storing each.
01414         //
01415 
01416         pui16Src = (uint16_t *)pui32SrcColors;
01417         while(ui32Count)
01418         {
01419             pui16Pal[ui32Start] = pui16Src[ui32Start];
01420             ui32Start++;
01421             ui32Count--;
01422         }
01423     }
01424 
01425     //
01426     // Write the pixel type into the first palette entry.
01427     //
01428     pui16Pal[0] &= ~(LCD_PALETTE_TYPE_8BPP | LCD_PALETTE_TYPE_DIRECT);
